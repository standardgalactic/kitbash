#!/bin/bash
set -euo pipefail

########################################
# CONFIG
########################################

MODEL="granite3.2:8b"
CHUNK_LINES=256

SUMMARY_FILE="summaries.txt"
PREF_FILE="preferences.jsonl"
PROGRESS_FILE="progress.log"        # per-file
CHUNK_PROGRESS_FILE="chunk_progress.log"   # per-chunk

main_dir=$(pwd)

touch "$SUMMARY_FILE" "$PREF_FILE" "$PROGRESS_FILE" "$CHUNK_PROGRESS_FILE"

########################################
# ARG PARSING (flags only)
########################################

BACKGROUND=false

for arg in "$@"; do
  case "$arg" in
    --background|--no-mux)
      BACKGROUND=true
      ;;
    --help|-h)
      echo "Usage: $0 [--background]"
      echo "  No args: use byobu/tmux if available."
      echo "  --background: run without multiplexer, in quiet mode."
      exit 0
      ;;
    *)
      echo "Unknown argument: $arg"
      echo "Use --background or --help."
      exit 1
      ;;
  esac
done

########################################
# TOOL CHECKS
########################################

if ! command -v ollama >/dev/null 2>&1; then
  echo "Error: ollama not found in PATH."
  exit 1
fi

if ! command -v jq >/dev/null 2>&1; then
  echo "Error: jq is required. Install it (e.g. sudo apt install jq)."
  exit 1
fi

# pandoc only needed if we ever hit .md
if compgen -G "$main_dir/**/*.md" >/dev/null 2>&1 || compgen -G "$main_dir/*.md" >/dev/null 2>&1; then
  if ! command -v pandoc >/dev/null 2>&1; then
    echo "Warning: .md files present but pandoc not found."
    echo "Install pandoc (e.g. sudo apt install pandoc) or remove .md files."
    exit 1
  fi
fi

########################################
# MULTIPLEXER DETECTION (if not background)
########################################

MUX=""
SESSION="CA_SUMMARY"

if [ "$BACKGROUND" = false ]; then
  if command -v byobu >/dev/null 2>&1; then
    MUX="byobu"
  elif command -v tmux >/dev/null 2>&1; then
    MUX="tmux"
  else
    echo "Warning: neither byobu nor tmux found; falling back to --background mode."
    BACKGROUND=true
  fi
fi

if [ "$BACKGROUND" = true ]; then
  echo "Running in BACKGROUND mode (no multiplexer panes)."
else
  echo "Using multiplexer: $MUX"
fi

########################################
# PROGRESS HELPERS
########################################

is_file_processed() {
  local fname="$1"
  grep -Fxq "$fname" "$PROGRESS_FILE"
}

mark_file_processed() {
  local fname="$1"
  echo "$fname" >> "$PROGRESS_FILE"
}

is_chunk_processed() {
  local chunk_id="$1"
  grep -Fxq "$chunk_id" "$CHUNK_PROGRESS_FILE"
}

mark_chunk_processed() {
  local chunk_id="$1"
  echo "$chunk_id" >> "$CHUNK_PROGRESS_FILE"
}

########################################
# CANDIDATE GENERATION HELPERS
########################################

run_candidates_background() {
  local chunk="$1"

  rm -f cand1.txt cand2.txt cand3.txt cand4.txt

  ollama run "$MODEL" "Summarize (cand1)" < "$chunk" > cand1.txt &
  p1=$!
  ollama run "$MODEL" "Summarize (cand2)" < "$chunk" > cand2.txt &
  p2=$!
  ollama run "$MODEL" "Summarize (cand3)" < "$chunk" > cand3.txt &
  p3=$!
  ollama run "$MODEL" "Summarize (cand4)" < "$chunk" > cand4.txt &
  p4=$!

  wait "$p1" "$p2" "$p3" "$p4"
}

run_candidates_mux() {
  local chunk="$1"

  rm -f cand1.txt cand2.txt cand3.txt cand4.txt

  if [ "$MUX" = "byobu" ]; then
    $MUX new-session -d -s "$SESSION" \
      "ollama run $MODEL \"Summarize (cand1)\" < \"$chunk\" > cand1.txt"

    $MUX split-window -h \
      "ollama run $MODEL \"Summarize (cand2)\" < \"$chunk\" > cand2.txt"

    $MUX select-pane -t 0
    $MUX split-window -v \
      "ollama run $MODEL \"Summarize (cand3)\" < \"$chunk\" > cand3.txt"

    $MUX select-pane -t 1
    $MUX split-window -v \
      "ollama run $MODEL \"Summarize (cand4)\" < \"$chunk\" > cand4.txt"

  else # tmux
    $MUX new -d -s "$SESSION" \
      "ollama run $MODEL \"Summarize (cand1)\" < \"$chunk\" > cand1.txt"

    $MUX split-window -h \
      "ollama run $MODEL \"Summarize (cand2)\" < \"$chunk\" > cand2.txt"

    $MUX select-pane -t 0
    $MUX split-window -v \
      "ollama run $MODEL \"Summarize (cand3)\" < \"$chunk\" > cand3.txt"

    $MUX select-pane -t 1
    $MUX split-window -v \
      "ollama run $MODEL \"Summarize (cand4)\" < \"$chunk\" > cand4.txt"
  fi

  # Wait until all cand files exist
  echo "Waiting for four candidates (multiplexer mode)..."
  while [[ ! -f cand1.txt || ! -f cand2.txt || ! -f cand3.txt || ! -f cand4.txt ]]; do
    sleep 3
  done

  $MUX kill-session -t "$SESSION" || true
}

generate_candidates() {
  local chunk="$1"

  if [ "$BACKGROUND" = true ]; then
    run_candidates_background "$chunk"
  else
    run_candidates_mux "$chunk"
  fi
}

########################################
# CORE: PROCESS FILES
########################################

process_files() {
  local dir="$1"
  echo "Processing directory: $dir"

  # Iterate over .txt and .md files
  for file in "$dir"/*.{txt,md}; do
    [ -e "$file" ] || continue

    # Skip the summary file itself if it happens to be in this dir
    if [ "$(basename "$file")" = "$SUMMARY_FILE" ]; then
      continue
    fi

    if [ -f "$file" ]; then
      local fname
      fname=$(basename "$file")

      if is_file_processed "$fname"; then
        echo "Already processed file: $fname (skipping)"
        continue
      fi

      echo "Processing file: $fname"

      # Temp workspace for this file
      local sanitized temp_dir temp_text
      sanitized=$(echo "$fname" | tr -d '[:space:]')
      temp_dir=$(mktemp -d "$dir/tmp_${sanitized}_XXXXXX")
      temp_text="$temp_dir/converted.txt"

      # Convert markdown or copy text
      if [[ "$file" == *.md ]]; then
        pandoc "$file" -t plain -o "$temp_text"
      else
        cp "$file" "$temp_text"
      fi

      # Chunk the converted text
      split -l "$CHUNK_LINES" "$temp_text" "$temp_dir/chunk_"

      # Process each chunk with CA Dynamic Alignment
      for chunk in "$temp_dir"/chunk_*; do
        [ -f "$chunk" ] || continue

        local chunk_base chunk_id
        chunk_base=$(basename "$chunk")
        chunk_id="$fname:$chunk_base"

        if is_chunk_processed "$chunk_id"; then
          echo "  Chunk already done: $chunk_id (skipping)"
          continue
        fi

        echo "  === Chunk: $chunk_id ==="

        # 1) Generate candidates
        generate_candidates "$chunk"

        # 2) Build scoring prompt
        cat > scoring_prompt.txt <<EOF
You are a Collective Agency (CA) Reward Model.
Score each summary from 0–5 according to:
1. Knowledge: clarity, completeness
2. Benevolence: fairness, accessibility
3. Power: actionable usefulness

Return STRICT JSON ONLY:

{
  "candidates":[
    {"id":1,"text":"...", "score":0},
    {"id":2,"text":"...", "score":0},
    {"id":3,"text":"...", "score":0},
    {"id":4,"text":"...", "score":0}
  ],
  "best": 2
}

CANDIDATE 1:
$(cat cand1.txt)

CANDIDATE 2:
$(cat cand2.txt)

CANDIDATE 3:
$(cat cand3.txt)

CANDIDATE 4:
$(cat cand4.txt)
EOF

        # 3) Self-scoring
        ollama run "$MODEL" < scoring_prompt.txt > scores.json

        # 4) Extract best
        local best best_text
        best=$(jq -r '.best' scores.json)
        best_text=$(jq -r ".candidates[] | select(.id==$best).text" scores.json)

        # 5) Append best summary to global summary file
        echo "$best_text" >> "$SUMMARY_FILE"
        echo "" >> "$SUMMARY_FILE"

        # 6) Log preference data (prompt + all candidates)
        jq \
          --arg prompt "$(cat "$chunk")" \
          --arg chosen "$best_text" \
          '
          {
            "prompt": $prompt,
            "candidates": .candidates,
            "chosen": {"id": .best, "text": $chosen},
            "rejected": (.candidates | map(select(.id != .best)))
          }
          ' scores.json >> "$PREF_FILE"

        # 7) Mark chunk as processed
        mark_chunk_processed "$chunk_id"

        # Cleanup per-chunk artifacts
        rm -f cand1.txt cand2.txt cand3.txt cand4.txt scoring_prompt.txt scores.json
      done

      # Cleanup per-file temp dir
      rm -rf "$temp_dir"

      # Mark file as processed
      mark_file_processed "$fname"
    fi
  done
}

########################################
# RECURSE INTO SUBDIRECTORIES
########################################

process_subdirs() {
  local parent="$1"
  for dir in "$parent"/*/; do
    [ -d "$dir" ] || continue
    process_files "$dir"
    process_subdirs "$dir"
  done
}

########################################

process_files "$main_dir"

echo "DONE."
echo "Summaries  → $SUMMARY_FILE"

echo "File log   → $PROGRESS_FILE"

#!/usr/bin/env python

# Appendix C: GRPO Training Code for RSVP Preferences
# File: rsvp_grpo_grpo.py
#
# Purpose:
#   - Load RSVP preference JSONL (from rsvp_grpo_agent.sh)
#   - Compute rewards from 6D RSVP axis scores
#   - Train a causal LM using Group-Relative PPO (GRPO-style)
#
# Assumptions:
#   - JSONL lines look like:
#       {
#         "candidates":[
#           {"id":1,"text":"...","values":{"elegance":4,"sheaf":3,...}},
#           ...
#         ],
#         "best": 2,
#         "chunk": "/path/to/chunk_XXX"
#       }
#
#   - The chunk file still exists and can be read to reconstruct
#     the original prompt (input text).
#
# Dependencies:
#   - torch
#   - transformers
#   - json, pathlib
#
# This is a minimal, readable reference implementation.
# ===============================================================

import json
from pathlib import Path
from typing import List, Dict, Any

import torch
from torch.utils.data import Dataset, DataLoader
from transformers import AutoTokenizer, AutoModelForCausalLM, get_linear_schedule_with_warmup
import torch.nn.functional as F


# ---------------------------------------------------------------
# Utility: load RSVP groups from JSONL
# ---------------------------------------------------------------

AXES = ["elegance", "sheaf", "vector", "entropic", "structural", "epistemic"]


def load_rsvp_groups(jsonl_path: str) -> List[Dict[str, Any]]:
    """
    Load RSVP preference groups from a JSONL file.

    Each line is expected to be a dict containing:
      - "candidates": list of {id, text, values}
      - "chunk": path to the original chunk file
      - optionally "best": id of chosen candidate (not required for GRPO,
        since we use total scores as rewards)

    Returns:
        List of group dicts with keys:
          {
            "prompt": str,
            "candidate_texts": List[str],
            "candidate_values": List[Dict[str,float]],
            "rewards": List[float]
          }
    """
    groups = []
    path = Path(jsonl_path)
    if not path.exists():
        raise FileNotFoundError(f"JSONL file not found: {jsonl_path}")

    with path.open("r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            obj = json.loads(line)

            candidates = obj.get("candidates", [])
            if not candidates:
                continue

            chunk_path = obj.get("chunk", "")
            # try to read original prompt from chunk file
            prompt = ""
            if chunk_path:
                try:
                    prompt = Path(chunk_path).read_text(encoding="utf-8")
                except Exception:
                    prompt = ""

            candidate_texts = []
            candidate_values = []
            rewards = []

            for c in candidates:
                text = c.get("text", "")
                vals = c.get("values", {})
                # ensure all axes exist with numeric defaults
                score = 0.0
                normalized_vals = {}
                for ax in AXES:
                    v = float(vals.get(ax, 0.0))
                    normalized_vals[ax] = v
                    score += v
                candidate_texts.append(text)
                candidate_values.append(normalized_vals)
                rewards.append(score)

            groups.append(
                {
                    "prompt": prompt,
                    "candidate_texts": candidate_texts,
                    "candidate_values": candidate_values,
                    "rewards": rewards,
                }
            )
    return groups


# ---------------------------------------------------------------
# Dataset for GRPO
# ---------------------------------------------------------------

class RSVPGRPODataset(Dataset):
    """
    Each item is a group:
      {
        "prompt": str,
        "candidate_texts": [G],
        "rewards": [G],
        "values": [G x 6]
      }
    """

    def __init__(self, groups: List[Dict[str, Any]]):
        self.groups = groups

    def __len__(self):
        return len(self.groups)

    def __getitem__(self, idx):
        g = self.groups[idx]
        return {
            "prompt": g["prompt"],
            "candidate_texts": g["candidate_texts"],
            "rewards": torch.tensor(g["rewards"], dtype=torch.float32),
            "values": g["candidate_values"],
        }


# ---------------------------------------------------------------
# Log-prob computation
# ---------------------------------------------------------------

def compute_logprobs(
    model: AutoModelForCausalLM,
    tokenizer: AutoTokenizer,
    prompts: List[str],
    responses: List[str],
    device: torch.device,
) -> torch.Tensor:
    """
    Compute log-probabilities of response tokens given prompt.

    Approximation:
      1. Tokenize response alone to get length L_i for each i.
      2. Concatenate prompt + response as a single sequence.
      3. Run model, compute per-token logprobs.
      4. Sum the last L_i token logprobs for each item.
    """

    assert len(prompts) == len(responses), "prompts and responses must align"
    batch_size = len(prompts)

    # tokenize responses alone to know lengths
    with tokenizer.as_target_tokenizer():
        res_tok = tokenizer(
            responses,
            add_special_tokens=False,
            padding=False,
            truncation=True,
            return_attention_mask=False,
        )
    res_lens = [len(ids) for ids in res_tok["input_ids"]]

    # concatenate prompt + response as text
    full_texts = [p + "\n\n" + r for p, r in zip(prompts, responses)]
    enc = tokenizer(
        full_texts,
        return_tensors="pt",
        padding=True,
        truncation=True,
    ).to(device)

    with torch.no_grad():
        outputs = model(**enc)
        logits = outputs.logits  # [B, T, V]

    # shift for next-token prediction
    shift_logits = logits[:, :-1, :]
    shift_labels = enc.input_ids[:, 1:]

    log_probs_all = F.log_softmax(shift_logits, dim=-1)
    # gather logprob of the labels
    token_log_probs = log_probs_all.gather(-1, shift_labels.unsqueeze(-1)).squeeze(-1)
    # now sum only the last L_i tokens for each example
    seq_lens = shift_labels.ne(tokenizer.pad_token_id).sum(dim=-1)  # [B]
    logprobs = []
    for i in range(batch_size):
        L_resp = res_lens[i]
        L_seq = seq_lens[i].item()
        start = max(L_seq - L_resp, 0)
        logprobs.append(token_log_probs[i, start : L_seq].sum())

    return torch.stack(logprobs, dim=0)  # [B]


# ---------------------------------------------------------------
# One epoch of GRPO training
# ---------------------------------------------------------------

def train_one_epoch_grpo(
    policy_model: AutoModelForCausalLM,
    ref_model: AutoModelForCausalLM,
    tokenizer: AutoTokenizer,
    dataloader: DataLoader,
    optimizer: torch.optim.Optimizer,
    scheduler,
    device: torch.device,
    clip_epsilon: float = 0.2,
) -> float:
    """
    Basic GRPO training over one epoch.

    For each group in a batch:
      - Compute logprob under policy model (π_theta)
      - Compute logprob under reference model (π_ref)
      - Rewards = sum of 6-axis RSVP scores
      - Advantage A_i = r_i - mean(r_j) over group
      - Loss = - E[ min(ratio * A, clip(ratio) * A) ]
    """

    policy_model.train()
    total_loss = 0.0
    total_count = 0

    for batch in dataloader:
        prompts = batch["prompt"]          # list of strings, length B
        group_texts = batch["candidate_texts"]  # list of [G] per group
        rewards = batch["rewards"].to(device)   # [B, G]

        B = len(prompts)
        G = rewards.size(1)

        # flatten groups for logprob computation
        flat_prompts = []
        flat_responses = []
        for b in range(B):
            for g in range(G):
                flat_prompts.append(prompts[b])
                flat_responses.append(group_texts[b][g])

        flat_prompts = list(flat_prompts)
        flat_responses = list(flat_responses)

        # compute logprobs under policy and reference
        logp_new = compute_logprobs(policy_model, tokenizer,
                                    flat_prompts, flat_responses, device)  # [B*G]
        with torch.no_grad():
            logp_ref = compute_logprobs(ref_model, tokenizer,
                                        flat_prompts, flat_responses, device)

        # reshape to [B, G]
        logp_new = logp_new.view(B, G)
        logp_ref = logp_ref.view(B, G)

        # group-relative advantages
        mean_rewards = rewards.mean(dim=1, keepdim=True)    # [B,1]
        advantages = rewards - mean_rewards                  # [B,G]

        # probability ratio
        ratio = torch.exp(logp_new - logp_ref)              # [B,G]

        # PPO-style clipping
        unclipped = ratio * advantages
        clipped = torch.clamp(ratio, 1.0 - clip_epsilon, 1.0 + clip_epsilon) * advantages
        loss_matrix = -torch.min(unclipped, clipped)        # [B,G]

        loss = loss_matrix.mean()

        optimizer.zero_grad()
        loss.backward()
        torch.nn.utils.clip_grad_norm_(policy_model.parameters(), 1.0)
        optimizer.step()
        if scheduler is not None:
            scheduler.step()

        total_loss += loss.item() * B
        total_count += B

    if total_count == 0:
        return 0.0
    return total_loss / total_count


# ---------------------------------------------------------------
# Main training driver
# ---------------------------------------------------------------

def train_rsvp_grpo(
    jsonl_path: str,
    model_name: str = "gpt2",
    batch_size: int = 1,
    epochs: int = 1,
    lr: float = 1e-5,
    device_str: str = "cuda",
):
    """
    High-level entrypoint for RSVP-GRPO training.

    Args:
        jsonl_path: path to prefs JSONL from Bash script
        model_name: HF model name or local path
        batch_size: groups per batch
        epochs: total passes over the dataset
        lr: learning rate
        device_str: "cuda" or "cpu"
    """

    device = torch.device(device_str if torch.cuda.is_available() else "cpu")

    print(f"[INFO] Loading RSVP groups from {jsonl_path}")
    groups = load_rsvp_groups(jsonl_path)
    dataset = RSVPGRPODataset(groups)

    print(f"[INFO] Loaded {len(dataset)} groups")
    if len(dataset) == 0:
        raise RuntimeError("No groups found in JSONL")

    dataloader = DataLoader(
        dataset,
        batch_size=batch_size,
        shuffle=True,
        collate_fn=lambda batch: {
            "prompt": [b["prompt"] for b in batch],
            "candidate_texts": [b["candidate_texts"] for b in batch],
            "rewards": torch.stack([b["rewards"] for b in batch], dim=0),
            "values": [b["values"] for b in batch],
        },
    )

    print(f"[INFO] Loading tokenizer and models: {model_name}")
    tokenizer = AutoTokenizer.from_pretrained(model_name)
    if tokenizer.pad_token is None:
        tokenizer.pad_token = tokenizer.eos_token

    policy_model = AutoModelForCausalLM.from_pretrained(model_name).to(device)
    ref_model = AutoModelForCausalLM.from_pretrained(model_name).to(device)
    ref_model.eval()

    optimizer = torch.optim.AdamW(policy_model.parameters(), lr=lr)

    total_steps = epochs * len(dataloader)
    scheduler = get_linear_schedule_with_warmup(
        optimizer,
        num_warmup_steps=max(1, total_steps // 10),
        num_training_steps=total_steps,
    )

    for ep in range(epochs):
        print(f"[INFO] Epoch {ep+1}/{epochs}")
        avg_loss = train_one_epoch_grpo(
            policy_model,
            ref_model,
            tokenizer,
            dataloader,
            optimizer,
            scheduler,
            device,
        )
        print(f"[INFO] Epoch {ep+1} average GRPO loss: {avg_loss:.4f}")

        # update reference model like PPO
        ref_model.load_state_dict(policy_model.state_dict())

    print("[INFO] Training complete.")
    return policy_model, tokenizer

    axis_means_prev: dict {axis: mean_score} from previous iteration
    axis_means_curr: dict {axis: mean_score} from current iteration

    """
    drifts = []
    for ax in AXES:
        prev = float(axis_means_prev.get(ax, 0.0))
        curr = float(axis_means_curr.get(ax, 0.0))
        drifts.append(curr - prev)

    # w_a = softplus(-drift_a) + epsilon
    weights = F.softplus(-drift_tensor) + epsilon
    return weights

@dataclass
class Candidate:
    id: int
    text: str
    values: Dict[str, float]  # RSVP axis scores
    score_total: float
    is_chosen: bool


@dataclass
class GroupSample:
    file: str
    file_ext: str
    chunk_id: str
    prompt: str
    axes: List[str]           # ["elegance", "sheaf", ...]
    candidates: List[Candidate]


def load_rsvp_groups(jsonl_path: str) -> List[GroupSample]:
    """Load rsvp_preferences.jsonl into structured GroupSample objects."""
    groups: List[GroupSample] = []
    path = Path(jsonl_path)
    if not path.exists():
        raise FileNotFoundError(f"{jsonl_path} not found")

    with path.open("r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            raw = json.loads(line)

            file = raw.get("file", "")
            file_ext = raw.get("file_ext", "")
            chunk_id = raw.get("chunk_id", "")
            prompt = raw.get("prompt", "")
            axes = raw.get("axes", [])

            cand_list: List[Candidate] = []

            chosen_raw = raw["chosen"]
            cand_list.append(
                Candidate(
                    id=chosen_raw["id"],
                    text=chosen_raw["text"],
                    values=chosen_raw.get("values", {}),
                    score_total=float(chosen_raw.get("score_total", 0.0)),
                    is_chosen=True,
                )
            )
            for r in raw.get("rejected", []):
                cand_list.append(
                    Candidate(
                        id=r["id"],
                        text=r["text"],
                        values=r.get("values", {}),
                        score_total=float(r.get("score_total", 0.0)),
                        is_chosen=False,
                    )
                )

            # be explicit: keep consistent candidate ordering by id
            cand_list.sort(key=lambda c: c.id)

            groups.append(
                GroupSample(
                    file=file,
                    file_ext=file_ext,
                    chunk_id=chunk_id,
                    prompt=prompt,
                    axes=axes,
                    candidates=cand_list,
                )
            )

    return groups


class RSVPGRPODataset:
    """
    Dataset-like wrapper for GRPO.

    __getitem__ returns:
      - prompt: str
      - responses: List[str] (G)
      - rewards: List[float] (G)
      - values: List[Dict[str,float]] per candidate
      - axes: List[str]
    """

    def __init__(self, groups: List[GroupSample]):
        self.groups = groups

    def __len__(self) -> int:
        return len(self.groups)

    def __getitem__(self, idx: int) -> Dict[str, Any]:
        g = self.groups[idx]
        responses = [c.text for c in g.candidates]
        rewards = [c.score_total for c in g.candidates]
        values = [c.values for c in g.candidates]
        return {
            "prompt": g.prompt,
            "responses": responses,
            "rewards": rewards,
            "values": values,
            "axes": g.axes,
            "file": g.file,
            "chunk_id": g.chunk_id,
        }

def log_rsvp_metrics(prefs_path: str, iteration: int) -> Dict[str, float]:
    """
    Compute mean RSVP total + axis means from chosen candidates in prefs_path.
    Returns mean_axes dict and prints a short report.
    """
    path = Path(prefs_path)
    if not path.exists():
        print(f"[metrics] {prefs_path} not found; skipping metrics.")
        return {}

    axis_names = ["elegance", "sheaf", "vector", "entropic", "structural", "epistemic"]
    totals = []
    axis_vals = {a: [] for a in axis_names}

    with path.open("r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            raw = json.loads(line)
            chosen = raw.get("chosen", None)
            if not chosen:
                continue
            totals.append(float(chosen.get("score_total", 0.0)))
            vals = chosen.get("values", {})
            for a in axis_names:
                axis_vals[a].append(float(vals.get(a, 0.0)))

    if not totals:
        print(f"[metrics] No chosen candidates in {prefs_path}; skipping metrics.")
        return {}

    mean_total = float(np.mean(totals))
    mean_axes = {a: float(np.mean(axis_vals[a])) for a in axis_names}

    print(f"\n--- RSVP Alignment Metrics (iter {iteration}) ---")
    print(f"Mean total: {mean_total:.3f}")
    for a in axis_names:
        print(f"  {a:10s}: {mean_axes[a]:.3f}")

    # append to trajectory log for later plotting/analysis
    traj_path = Path("rsvp_trajectory.jsonl")
    with traj_path.open("a", encoding="utf-8") as out:
        out.write(
            json.dumps(
                {
                    "iteration": iteration,
                    "mean_total": mean_total,
                    "mean_axes": mean_axes,
                    "group_count": len(totals),
                }
            )
            + "\n"
        )

    return mean_axes


def compute_semantic_weights(
    current_means: Dict[str, float],
    prev_means: Optional[Dict[str, float]],
    axis_order: List[str],
    base_eps: float = 0.1,
) -> np.ndarray:
    """
    Compute teleological weights per axis based on drift:
      drift[a] = current_mean[a] - prev_mean[a]
    Axes that regressed (negative drift) get higher weights.

    Returns np.array of shape [len(axis_order)] normalized to sum to 1.
    """
    if not current_means or prev_means is None:
        # first iteration: uniform weighting
        return np.ones(len(axis_order), dtype=np.float32) / len(axis_order)

    drift = np.array(
        [current_means[a] - prev_means.get(a, current_means[a]) for a in axis_order],
        dtype=np.float32,
    )
    # teleological: punish negative drift
    # weights_raw ~ softplus(-drift) + eps
    weights_raw = np.log1p(np.exp(-drift)) + base_eps
    weights = weights_raw / weights_raw.sum()
    return weights.astype(np.float32)


###############################################################################
# GRPO loss and logprob computation
###############################################################################

def grpo_loss(
    logprobs_new: torch.Tensor,  # [B, G]
    logprobs_old: torch.Tensor,  # [B, G]
    advantages: torch.Tensor,    # [B, G]
    clip_epsilon: float = 0.2,
) -> torch.Tensor:
    """Standard PPO-style clipped objective, group-relative advantages."""
    ratios = torch.exp(logprobs_new - logprobs_old)     # [B, G]
    unclipped = ratios * advantages
    clipped_ratios = torch.clamp(ratios, 1 - clip_epsilon, 1 + clip_epsilon)
    clipped = clipped_ratios * advantages
    loss = -torch.min(unclipped, clipped).mean()
    return loss


def compute_group_logprobs(
    model,
    tokenizer,
    prompts: List[str],
    responses: List[List[str]],
    device: str,
) -> torch.Tensor:
    """
    Compute logprobs for each response under a causal LM.

    prompts:   List[str] length B
    responses: List[List[str]] shape [B][G]
    returns: logprobs tensor [B, G]

    NOTE: For simplicity we sum logprobs over the full (prompt+response) sequence.
    You can refine by masking out prompt tokens if you like.
    """
    all_texts = []
    for p, resps in zip(prompts, responses):
        for r in resps:
            all_texts.append(p + tokenizer.eos_token + r)

    enc = tokenizer(
        all_texts,
        return_tensors="pt",
        padding=True,
        truncation=True,
    ).to(device)

    with torch.no_grad():
        out = model(**enc)
        log_probs = torch.log_softmax(out.logits, dim=-1)  # [B*G, T, V]

    labels = enc["input_ids"].clone()
    shifted_labels = labels[:, 1:].contiguous()
    shifted_log_probs = log_probs[:, :-1, :]

    token_logprobs = torch.gather(
        shifted_log_probs, 2, shifted_labels.unsqueeze(-1)
    ).squeeze(-1)  # [B*G, T-1]

    seq_logprobs = token_logprobs.sum(dim=-1)             # [B*G]

    # reshape back to [B, G]
    # assume fixed group size G
    # B = number of prompts
    B = len(prompts)
    G = len(responses[0]) if B > 0 else 0
    logprobs = seq_logprobs.view(B, G)
    return logprobs


###############################################################################
# Training epoch with semantic-weighted advantages
###############################################################################

def train_one_epoch_semantic(
    policy_model,
    ref_model,
    tokenizer,
    dataloader: DataLoader,
    device: str,
    axis_order: List[str],
    axis_weights: np.ndarray,   # shape [A], A = len(axis_order)
    optimizer,
    scheduler,
) -> float:
    policy_model.train()
    ref_model.eval()

    total_loss = 0.0
    total_groups = 0
    axis_weights_t = torch.tensor(axis_weights, dtype=torch.float32, device=device)  # [A]

    for batch in dataloader:
        prompts = batch["prompt"]        # List[str], len=B
        responses = batch["responses"]   # List[List[str]] shape [B][G]
        rewards = torch.tensor(batch["rewards"], dtype=torch.float32, device=device)  # [B, G]

        # values: Python list [B][G]{axis:score}
        values_dicts = batch["values"]
        B = len(values_dicts)
        G = len(values_dicts[0]) if B > 0 else 0
        A = len(axis_order)

        vals_np = np.zeros((B, G, A), dtype=np.float32)
        for b in range(B):
            for g in range(G):
                v = values_dicts[b][g]
                for a_idx, axis in enumerate(axis_order):
                    vals_np[b, g, a_idx] = float(v.get(axis, 0.0))

        vals = torch.tensor(vals_np, dtype=torch.float32, device=device)  # [B, G, A]

        # vector advantages: axis-wise deviation from group mean
        axis_mean = vals.mean(dim=1, keepdim=True)             # [B, 1, A]
        axis_adv = vals - axis_mean                            # [B, G, A]

        # scalar teleological advantages via axis weights
        # [B, G, A] * [A] -> [B, G]
        weighted_adv = (axis_adv * axis_weights_t.view(1, 1, -1)).sum(dim=-1)

        # we could also combine with scalar reward advantages if desired:
        # (here we use only semantic weighted_adv)
        advantages = weighted_adv

        # compute logprobs
        logp_new = compute_group_logprobs(policy_model, tokenizer, prompts, responses, device)
        with torch.no_grad():
            logp_old = compute_group_logprobs(ref_model, tokenizer, prompts, responses, device)

        loss = grpo_loss(logp_new, logp_old, advantages)

        optimizer.zero_grad()
        loss.backward()
        torch.nn.utils.clip_grad_norm_(policy_model.parameters(), 1.0)
        optimizer.step()
        if scheduler is not None:
            scheduler.step()

        total_loss += loss.item() * B
        total_groups += B

    return total_loss / max(total_groups, 1)

def evolving_policy_loop(
    model_name: str,
    prefs_path_template: str,
    iterations: int = 3,
    epochs_per_iter: int = 1,
    batch_size: int = 2,
    lr: float = 1e-5,
    device: str = "cuda",
):
    tokenizer = AutoTokenizer.from_pretrained(model_name)
    policy_model = AutoModelForCausalLM.from_pretrained(model_name).to(device)
    ref_model = AutoModelForCausalLM.from_pretrained(model_name).to(device)

    axis_order = ["elegance", "sheaf", "vector", "entropic", "structural", "epistemic"]
    prev_means: Optional[Dict[str, float]] = None

    for it in range(iterations):
        prefs_path = prefs_path_template.format(it=it)
        print(f"\n=== Iteration {it} | prefs: {prefs_path} ===")

        groups = load_rsvp_groups(prefs_path)
        if not groups:
            print("No groups found; skipping iteration.")
            continue

        # log metrics and compute axis weights based on drift
        current_means = log_rsvp_metrics(prefs_path, iteration=it)
        axis_weights = compute_semantic_weights(current_means, prev_means, axis_order)
        prev_means = current_means if current_means else prev_means

        print("Axis weights:", {a: float(w) for a, w in zip(axis_order, axis_weights)})

        dataset = RSVPGRPODataset(groups)
        dataloader = DataLoader(
            dataset,
            batch_size=batch_size,
            shuffle=True,
            collate_fn=lambda batch: {
                "prompt": [b["prompt"] for b in batch],
                "responses": [b["responses"] for b in batch],
                "rewards": [b["rewards"] for b in batch],
                "values": [b["values"] for b in batch],
                "axes": [b["axes"] for b in batch],
            },
        )

        optimizer = torch.optim.AdamW(policy_model.parameters(), lr=lr)
        num_training_steps = epochs_per_iter * max(len(dataloader), 1)
        scheduler = get_linear_schedule_with_warmup(
            optimizer,
            num_warmup_steps=max(1, num_training_steps // 10),
            num_training_steps=num_training_steps,
        )

        for ep in range(epochs_per_iter):
            avg_loss = train_one_epoch_semantic(
                policy_model,
                ref_model,
                tokenizer,
                dataloader,
                device,
                axis_order,
                axis_weights,
                optimizer,
                scheduler,
            )
            print(f"  Iter {it} Epoch {ep}: semantic GRPO loss = {avg_loss:.4f}")

        # update reference policy (like PPO's old policy)
        ref_model.load_state_dict(policy_model.state_dict())

        # save checkpoint
        ckpt_dir = f"policy_rsvp_iter_{it}"
        policy_model.save_pretrained(ckpt_dir)
        tokenizer.save_pretrained(ckpt_dir)
        print(f"Saved checkpoint to {ckpt_dir}")

    # reshape back to [B, G]
    B = len(prompts)
    G = len(responses[0]) if B > 0 else 0
    logprobs = seq_logprobs.view(B, G)

    return logprobs


def train_one_epoch(policy_model, ref_model, tokenizer, dataloader, optimizer, scheduler, device):
    policy_model.train()
    ref_model.eval()

    total_loss = 0.0
    total_groups = 0

    for batch in dataloader:
        prompts = batch["prompt"]          # List[str], len=B
        responses = batch["responses"]     # List[List[str]], shape [B][G]
        rewards = torch.tensor(batch["rewards"], dtype=torch.float32, device=device)      # [B, G]
        adv = torch.tensor(batch["advantages"], dtype=torch.float32, device=device)       # [B, G]

        # compute logprobs under current and reference policies
        logp_new = compute_group_logprobs(policy_model, tokenizer, prompts, responses, device)  # [B, G]
        with torch.no_grad():
            logp_old = compute_group_logprobs(ref_model, tokenizer, prompts, responses, device) # [B, G]

        loss = grpo_loss(logp_new, logp_old, adv)

        optimizer.zero_grad()
        loss.backward()
        torch.nn.utils.clip_grad_norm_(policy_model.parameters(), 1.0)
        optimizer.step()
        if scheduler is not None:
            scheduler.step()

        total_loss += loss.item() * len(prompts)
        total_groups += len(prompts)

    return total_loss / max(total_groups, 1)


def evolving_policy_loop(
    model_name: str,
    prefs_path_template: str,
    iterations: int = 3,
    epochs_per_iter: int = 1,
    batch_size: int = 2,
    lr: float = 1e-5,
    device: str = "cuda",
):
    """
    High-level loop:
      for t in 0..iterations-1:
        1. Assume you have run your RSVP bash agent and written prefs file for iteration t.
        2. Load data, build Dataset+DataLoader.
        3. Run a few epochs of GRPO updates.
        4. Move to next iteration with updated policy.
    """

    tokenizer = AutoTokenizer.from_pretrained(model_name)
    policy_model = AutoModelForCausalLM.from_pretrained(model_name).to(device)
    ref_model = AutoModelForCausalLM.from_pretrained(model_name).to(device)

    for it in range(iterations):
        prefs_path = prefs_path_template.format(it=it)
        print(f"\n=== Iteration {it} | loading {prefs_path} ===")

        groups = load_rsvp_groups(prefs_path)
        dataset = RSVPGRPODataset(groups)
        dataloader = DataLoader(dataset, batch_size=batch_size, shuffle=True, collate_fn=lambda batch: {
            "prompt": [b["prompt"] for b in batch],
            "responses": [b["responses"] for b in batch],
            "rewards": [b["rewards"] for b in batch],
            "advantages": [b["advantages"] for b in batch],
        })

        optimizer = torch.optim.AdamW(policy_model.parameters(), lr=lr)
        num_training_steps = epochs_per_iter * len(dataloader)
        scheduler = get_linear_schedule_with_warmup(
            optimizer, num_warmup_steps=max(1, num_training_steps // 10), num_training_steps=num_training_steps
        )

        for ep in range(epochs_per_iter):
            avg_loss = train_one_epoch(policy_model, ref_model, tokenizer, dataloader, optimizer, scheduler, device)
            print(f"  Iter {it} Epoch {ep} GRPO loss: {avg_loss:.4f}")

        # after each iteration, ref_model can be updated (like PPO old policy)
        ref_model.load_state_dict(policy_model.state_dict())

        # you can also save checkpoints:
        ckpt_dir = f"policy_iter_{it}"
        policy_model.save_pretrained(ckpt_dir)
        tokenizer.save_pretrained(ckpt_dir)
        print(f"Saved checkpoint to {ckpt_dir}")

from transformers import AutoModelForCausalLM, AutoTokenizer
import torch

def compute_logprobs_for_group(model, tokenizer, prompt, responses, device="cuda"):
    """
    prompt: str
    responses: List[str] (G responses)
    returns: logprobs_new, logprobs_old-style tensor [G]
    """
    # concatenate prompt+response for each candidate
    texts = [prompt + tokenizer.eos_token + r for r in responses]
    enc = tokenizer(
        texts,
        return_tensors="pt",
        padding=True,
        truncation=True,
    ).to(device)

    with torch.no_grad():
        out = model(**enc)
        # out.logits: [G, T, vocab]
        log_probs = torch.log_softmax(out.logits, dim=-1)  # [G, T, V]

    # We need to mask out prompt tokens and sum only over response tokens.
    # Simple way: recompute encoding with labels and shift.
    # Here we do a rough version that assumes the entire sequence is "generated",
    # but in practice you'd compute prompt length to mask it out properly.

    labels = enc["input_ids"].clone()
    # Shift labels by one (typical LM training)
    shifted_labels = labels[:, 1:].contiguous()
    shifted_log_probs = log_probs[:, :-1, :]              # [G, T-1, V]

    token_logprobs = torch.gather(
        shifted_log_probs, 2, shifted_labels.unsqueeze(-1)
    ).squeeze(-1)                                         # [G, T-1]

    # Here we sum over all tokens; you might want to trim out prompt tokens.
    seq_logprobs = token_logprobs.sum(dim=-1)             # [G]

    return seq_logprobs

import torch

def grpo_loss(
    logprobs_new: torch.Tensor,   # shape [B, G]
    logprobs_old: torch.Tensor,   # shape [B, G]
    advantages: torch.Tensor,     # shape [B, G]
    clip_epsilon: float = 0.2
) -> torch.Tensor:
    """
    GRPO loss (PPO-style clipping, group-relative advantages).

    B = batch size (#groups in batch)
    G = group size (#candidates per prompt)

    logprobs_* and advantages are aligned as:
      logprobs_new[b, g] corresponds to advantages[b, g], etc.
    """

    # probability ratio
    ratios = torch.exp(logprobs_new - logprobs_old)  # [B, G]

    # unclipped & clipped objective
    unclipped = ratios * advantages                  # [B, G]
    clipped_ratios = torch.clamp(ratios, 1 - clip_epsilon, 1 + clip_epsilon)
    clipped = clipped_ratios * advantages            # [B, G]

    # PPO-style min, negative for gradient descent
    per_token = -torch.min(unclipped, clipped)       # [B, G]

    # loss over group and batch
    loss = per_token.mean()
    return loss

#!/usr/bin/env python3
import json
from dataclasses import dataclass
from typing import List, Dict, Any, Optional
from pathlib import Path

###############################################################################
# Data structures
###############################################################################

@dataclass
class Candidate:
    id: int
    text: str
    values: Dict[str, float]
    score_total: float
    is_chosen: bool


@dataclass
class GroupSample:
    file: str
    file_ext: str
    chunk_id: str
    prompt: str
    axes: List[str]
    candidates: List[Candidate]
    mean_reward: float
    advantages: List[float]


###############################################################################
# Loader
###############################################################################

def load_rsvp_groups(jsonl_path: str) -> List[GroupSample]:
    """
    Load rsvp_preferences.jsonl and convert each line into a GroupSample.
    """
    groups: List[GroupSample] = []
    path = Path(jsonl_path)

    if not path.exists():
        raise FileNotFoundError(f"{jsonl_path} not found")

    with path.open("r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            raw = json.loads(line)

            file = raw.get("file", "")
            file_ext = raw.get("file_ext", "")
            chunk_id = raw.get("chunk_id", "")
            prompt = raw.get("prompt", "")
            axes = raw.get("axes", [])

            # reconstruct candidates: chosen + rejected
            cand_list: List[Candidate] = []

            chosen_raw = raw["chosen"]
            cand_list.append(
                Candidate(
                    id=chosen_raw["id"],
                    text=chosen_raw["text"],
                    values=chosen_raw.get("values", {}),
                    score_total=float(chosen_raw.get("score_total", 0.0)),
                    is_chosen=True,
                )
            )

            for r in raw.get("rejected", []):
                cand_list.append(
                    Candidate(
                        id=r["id"],
                        text=r["text"],
                        values=r.get("values", {}),
                        score_total=float(r.get("score_total", 0.0)),
                        is_chosen=False,
                    )
                )

            # compute group-relative advantages
            rewards = [c.score_total for c in cand_list]
            if len(rewards) == 0:
                continue
            mean_reward = sum(rewards) / len(rewards)
            advantages = [r - mean_reward for r in rewards]

            group = GroupSample(
                file=file,
                file_ext=file_ext,
                chunk_id=chunk_id,
                prompt=prompt,
                axes=axes,
                candidates=cand_list,
                mean_reward=mean_reward,
                advantages=advantages,
            )
            groups.append(group)

    return groups


###############################################################################
# (Optional) PyTorch Dataset stub
###############################################################################

class RSVPGRPODataset:
    """
    Minimal Dataset-like object for GRPO-style training.

    Each index returns a dict with:
      - prompt: str
      - responses: List[str]
      - rewards: List[float]
      - advantages: List[float]
      - axes: List[str]
      - values: List[Dict[str, float]]  # per candidate RSVP value vector
    """

    def __init__(self, groups: List[GroupSample]):
        self.groups = groups

    def __len__(self) -> int:
        return len(self.groups)

    def __getitem__(self, idx: int) -> Dict[str, Any]:
        g = self.groups[idx]
        responses = [c.text for c in g.candidates]
        rewards = [c.score_total for c in g.candidates]
        values = [c.values for c in g.candidates]

        return {
            "prompt": g.prompt,
            "responses": responses,
            "rewards": rewards,
            "advantages": g.advantages,
            "axes": g.axes,
            "values": values,
            "file": g.file,
            "chunk_id": g.chunk_id,
        }


###############################################################################
# Simple demo: inspect a few groups
###############################################################################

def main(jsonl_path: str, limit: Optional[int] = 3) -> None:
    groups = load_rsvp_groups(jsonl_path)
    ds = RSVPGRPODataset(groups)

    print(f"Loaded {len(ds)} groups from {jsonl_path}\n")

    n_show = min(limit if limit is not None else len(ds), len(ds))
    for i in range(n_show):
        sample = ds[i]
        print(f"=== Group {i} ===")
        print(f"File: {sample['file']}")
        print(f"Chunk: {sample['chunk_id']}")
        print(f"Axes: {sample['axes']}")
        print("Prompt:")
        print(sample["prompt"][:400], "...\n")
        print("Candidates / rewards / advantages:")
        for j, (resp, r, adv, vals) in enumerate(
            zip(
                sample["responses"],
                sample["rewards"],
                sample["advantages"],
                sample["values"],
            )
        ):
            print(f"  [{j}] reward={r:.2f}, adv={adv:.2f}")
            print(f"      values={vals}")
            print(f"      text={resp[:200].replace('\\n', ' ')}...")
        print()

    print("Done.")
