# Patch node_weight to correctly sum incident edge weights
import networkx as nx

def node_weight(G: nx.Graph, node, scheme: str = "strength") -> float:
    if scheme == "degree":
        return float(G.degree(node))
    elif scheme == "strength":
        total = 0.0
        for u, v, w in G.edges(node, data="weight", default=1.0):
            total += float(w)
        return total
    elif scheme == "uniform":
        return 1.0
    else:
        # default to strength
        total = 0.0
        for u, v, w in G.edges(node, data="weight", default=1.0):
            total += float(w)
        return total
