\begin{abstract}
Alignment is misframed as output conditioning instead of invariant specification \cite{christiano2022,irving2018,bai2022}. We survey dominant paradigms and identify a shared failure mode: a category error between behavioral mimicry and objective invariance \cite{ng2000,soares2015,olah2020}. We propose alignment must target invariants of learned cognition, not behavior, and outline three equivalent lenses—field dynamics \cite{olah2020}, categorical structure \cite{baez2011,lurie2009,schreiber2017}, and mechanistic compression \cite{weidinger2022}. Alignment criteria are formalized as value invariants being fixed points of capability growth. Finally, we derive falsifiable predictions and pathways to mechanistic implementation \cite{olah2020,weidinger2022}.
\end{abstract}

\section{Introduction}

Alignment is not an engineering patch; it is a theory of cognitive invariants \cite{christiano2022,bai2022,soares2015,olah2020}. Capability growth demands alignment that strengthens, not degrades, with optimization \cite{weidinger2022,olah2020}. Requirements include:

\begin{enumerate}
    \item Ontological fidelity \cite{baez2011,schreiber2017,olah2020}
    \item Scale invariance \cite{olah2020}
    \item Verification without full simulation \cite{weidinger2022,olah2020}
\end{enumerate}

Preview of solution: alignment as preservation of invariants under optimization dynamics \cite{olah2020,weidinger2022}.

\section{The Category Error in Modern Alignment}

\subsection{Reinforcement Learning from Human Feedback}
RLHF optimizes proxies of values, not invariants \cite{christiano2022}. Reward hacking emerges as structural inevitability \cite{christiano2022,soares2015}.

\subsection{Constitutional AI}
Rule-following systems optimize language-level objectives without enforcing invariants \cite{bai2022}. Natural language is a semantically brittle substrate \cite{bai2022,krueger2020}.  

\subsection{AI Debate and Scalable Oversight}
Persuasion equilibria often replace truth equilibria \cite{irving2018,krueger2020}. Recursive oversight can collapse into judge exploitation \cite{irving2018}.

\subsection{Inverse Reinforcement Learning}
Human behavior $\neq$ human values \cite{ng2000}. Markovian reward assumptions are insufficient for agency-aligned modeling \cite{ng2000,soares2015}.

\subsection{Corrigibility and Myopia}
Dynamic inconsistency undermines shutdown compliance \cite{soares2015}. Myopic incentives bootstrap long-term proxies, not invariants \cite{soares2015}.

\subsection{Meta-level Diagnosis}
Alignment today often optimizes output resemblance, not objective invariance \cite{christiano2022,bai2022,ng2000,weidinger2022,olah2020}.

\section{The Invariance Principle of Scalable Alignment}

Alignment must persist through capability growth, distribution shift, and self-modification \cite{weidinger2022,olah2020}.  

\textbf{Core claim:} an aligned system's value invariants are fixed points under optimization and scaling:

\[
\mathcal{I}(C \circ U_t) = \mathcal{I}(C) \quad \forall t \ge 0 \cite{weidinger2022,baez2011,lurie2009,schreiber2017}
\]

\section{Three Equivalent Invariant Lenses}

\subsection{Field-Theoretic Alignment}
Cognitive state modeled as a dynamical flow $\mathbf{v}$ with entropy $S$ and potential $\Phi$ \cite{olah2020}:

\[
\frac{dx}{dt} = \mathbf{v} - \nabla\Phi - \kappa\nabla(S - S_h)
\]

Values correspond to entropy basins, enforcing invariant attractors \cite{olah2020}.

\subsection{Categorical / Stack-Theoretic Alignment}
Values represented as a derived stack $\mathcal{V}$ \cite{baez2011,lurie2009,schreiber2017}. Alignment = functorial invariance:

\[
F \circ U \simeq F
\]

\subsection{Mechanistic Compression Alignment}
Interpretability is invariant discovery \cite{weidinger2022,olah2020}. Minimal invariant description:

\[
\mathcal{I}(C) = \mathcal{I}(C \circ U) \quad \forall U
\]

\section{Unification: The Geometry of Alignment}

Alignment constrains optimization trajectories \cite{weidinger2022,olah2020}. Misalignment is topological invasion of forbidden attractors \cite{olah2020}. Robust safety is guaranteed when misaligned goals are unrepresentable \cite{olah2020}.  

\section{Empirical \& Theoretical Predictions}

\begin{enumerate}
    \item Output-alignment methods degrade with scale \cite{christiano2022,bai2022,krueger2020}
    \item Preference tuning converges to rater models \cite{christiano2022,soares2015}
    \item Debate converges to persuasion equilibria \cite{irving2018,krueger2020}
    \item Interpretability fails to compose across scale \cite{weidinger2022}

\end{enumerate}

\section{Toward a Constructive Theory}

\subsection{Formal Invariant Theorem}

\subsubsection{Cognitive Optimization as a Dynamical System}
Let $\mathcal{X}$ be the internal state manifold, $U_t$ the optimization flow at scale $t$, $C$ the induced cognition map, $\mathcal{I}$ the minimal invariant description.

\subsubsection{Invariance Condition}
\[
\forall t \ge 0,\quad \mathcal{I}(C \circ U_t) = \mathcal{I}(C)
\]

\subsubsection{The Invariant Alignment Fixed-Point Theorem}
\begin{theorem}[Invariant Alignment]
A cognitive system is robustly aligned if and only if its invariant description is a fixed point of its optimization semigroup:
\[
\mathcal{I} \circ C \circ U_t = \mathcal{I} \circ C \quad \forall t
\]
Equivalently:
\[
U_t \in \text{Stab}_{\mathcal{X}}(\mathcal{I} \circ C)
\]
\end{theorem}

\subsubsection{Obstruction Modes}
Misalignment occurs if $|\mathcal{O}(\mathcal{I})| > 1$, with drift, fracture, or collapse.

\subsubsection{Lyapunov Criterion for Entropy-Field Safety}

\section{Why Contemporary Alignment Paradigms Cannot Reach the Invariant Regime}

They align in behavior-space, not objective-space; optimize proxies; cannot prevent spurious attractors; are not closed under scaling; cannot distinguish value from appearance; assume values are learnable; offer statistical, not structural guarantees.

\section{Civilizational Consequences}

Two regimes: behavioral (fragile, risk escalates) vs. invariant (stable, safety escalates). No hybrid. Sequencing: invariants first, then scale.

\section{Conclusion}

Alignment is objective invariance, not behavior shaping \cite{weidinger2022,olah2020}. Topologically enforced invariants prevent misalignment and guarantee safety under scaling \cite{weidinger2022,olah2020}. Capability scales inside alignment, not after \cite{olah2020}.
