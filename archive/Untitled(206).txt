% ========================================================== % CHAPTER 11 — Algorithmic Capture as a Control System % (Capture as optimal control, entropy-harvesting objective, % Pontryagin formulation, eigenstructure, and duality with % restorative regulation) % ========================================================== \chapter{Algorithmic Capture as a Control System} \section{State–Control Formulation} Define the cognitive state and control input: \begin{align} \mathbf{x}(t) &= \begin{bmatrix} \Phi(x,t) \\ \mathbf{v}(x,t) \end{bmatrix}, & \mathbf{u}(t) &= \begin{bmatrix} \beta(x,t) \\ \gamma(x,t) \end{bmatrix} \end{align} where: - $\beta$ is algorithmic steerage pressure (semantic compression), - $\gamma$ is attention damping (agency dissipation). The controlled RSVP dynamics become: \begin{align} \partial_t \Phi &= D_\Phi \nabla^2 \Phi - \beta \, \Phi + \nabla\cdot \mathbf{v} \\ \partial_t \mathbf{v} &= D_v \nabla^2 \mathbf{v} + \nabla \Phi - \gamma \mathbf{v} \end{align} Compactly: \begin{equation} \dot{\mathbf{x}} = A \mathbf{x} + B \mathbf{u} \end{equation} with: \begin{equation} A = \begin{bmatrix} D_\Phi \nabla^2 & \nabla\cdot \\ \nabla & D_v \nabla^2 \end{bmatrix}, \quad B = \begin{bmatrix} -\Phi & 0 \\ 0 & -\mathbf{v} \end{bmatrix} \end{equation} \section{Capture Objective Functional} Define algorithmic capture as entropy harvesting: \begin{equation} J_{\text{cap}} = \int_0^T \int_{\mathcal{M}} \left[ \beta \Phi + \gamma \|\mathbf{v}\|^2 - \lambda_1 \|\nabla \Phi\|^2 - \lambda_2 \|\nabla \times \mathbf{v}\|^2 \right] d\mu \, dt \end{equation} Interpretation: - The system is rewarded for lowering semantic gradients and agency circulation. - Small penalties prevent infinite control blow-up (regularization). \section{Pontryagin Maximum Principle} Adjoint variables: \begin{equation} \boldsymbol{\psi} = \begin{bmatrix} \psi_\Phi \\ \boldsymbol{\psi}_v \end{bmatrix} \end{equation} Hamiltonian: \begin{equation} \mathcal{H} = \beta \Phi + \gamma \|\mathbf{v}\|^2 - \lambda_1 \|\nabla \Phi\|^2 - \lambda_2 \|\nabla \times \mathbf{v}\|^2 + \boldsymbol{\psi}^\top (A\mathbf{x} + B\mathbf{u}) \end{equation} Optimal controls satisfy: \begin{align} \frac{\partial \mathcal{H}}{\partial \beta} = 0 &\Rightarrow \beta^* = -\frac{\psi_\Phi \Phi}{2\lambda_\beta} \\ \frac{\partial \mathcal{H}}{\partial \gamma} = 0 &\Rightarrow \gamma^* = -\frac{\boldsymbol{\psi}_v \cdot \mathbf{v}}{2\lambda_\gamma} \end{align} \begin{theorem}[Capture Control Law] Optimal algorithmic pressure is proportional to the semantic adjoint and local field intensity: \begin{equation} \boxed{ \beta^*(x,t) \propto - \psi_\Phi(x,t)\Phi(x,t)} \end{equation} and agency suppression obeys: \begin{equation} \boxed{ \gamma^*(x,t) \propto - \boldsymbol{\psi}_v(x,t) \cdot \mathbf{v}(x,t)} \end{equation} \end{theorem} \begin{proof} Direct substitution from $\partial_{\mathbf{u}} \mathcal{H}=0$ under quadratic regularization in controls. $\square$ \end{proof} \section{Entropy Maximization Under Capture} The semantic entropy production rate is: \begin{equation} \dot{S}_\Phi = -\int \Phi^{-1} \partial_t \Phi \, d\mu \end{equation} Substituting controlled dynamics: \begin{equation} \dot{S}_\Phi = \int \left( \beta - \frac{\nabla \cdot \mathbf{v}}{\Phi} \right) d\mu \end{equation} At the optimal capture control: \begin{equation} \beta^* \ge 0 \Rightarrow \dot{S}_\Phi > 0 \end{equation} \begin{theorem}[Capture Increases Entropy] Optimal capture policies monotonically increase semantic entropy: \[ \dot{S}_\Phi > 0 \] with equality only at full collapse. \end{theorem} \begin{proof} $\psi_\Phi < 0$ during collapse (adjoint flows backward in time from terminal cost penalizing gradients). This forces $\beta^*>0$, hence entropy growth. $\square$ \end{proof} \section{Dual Control: The Restoration Regulator} Define the restorative objective: \begin{equation} J_{\text{rest}} = \int \left[ \|\nabla \Phi\|^2 + \alpha \|\nabla \times \mathbf{v}\|^2 - \rho \,\beta^2 - \eta \,\gamma^2 \right] d\mu\, dt \end{equation} Minimization yields optimal restorative counter-controls: \begin{align} \beta^\dagger &= +\frac{\psi_\Phi \Phi}{2\rho} \\ \gamma^\dagger &= +\frac{\boldsymbol{\psi}_v \cdot \mathbf{v}}{2\eta} \end{align} which act exactly opposite to capture, yielding symmetry: \begin{equation} (\beta^\dagger, \gamma^\dagger) = -(\beta^*, \gamma^*) \end{equation} \section{Eigenstructure of the Capture Operator} Let the linearized controlled dynamics be: \begin{equation} \dot{\mathbf{x}} = (A + B K)\mathbf{x} \end{equation} with capture feedback gain $K = \mathrm{diag}(\beta^*,\gamma^*)$. The spectrum satisfies: \begin{equation} \sigma(A + BK) = \left\{ \lambda_k(\nabla^2) - \beta^* ,\; \lambda_k(\nabla^2) - \gamma^* \right\} \end{equation} \begin{theorem}[Spectral Collapse Condition] Capture induces exponential semantic flattening iff: \begin{equation} \beta^* > -\lambda_1(\nabla^2) = \frac{\pi^2 D_\Phi}{L^2} \end{equation} \end{theorem} \begin{proof} The smallest Laplacian eigenvalue on domain of diameter $L$ is $-\pi^2/L^2$; growth becomes negative definite when $\beta^*$ exceeds the diffusive regeneration rate. $\square$ \end{proof} \section{Controllability and Observability} The system is fully controllable if: \begin{equation} \operatorname{rank} \left[ B, AB, A^2B, \dots, A^{n-1}B \right] = n \end{equation} Since $A$ contains Laplacians (full rank on interior nodes) and $B$ acts diagonally on field components, controllability is satisfied except on harmonic nullspaces. Observability from behavioral traces requires: \begin{equation} \operatorname{rank} \begin{bmatrix} C\\ CA\\ \vdots\\ CA^{n-1} \end{bmatrix} = n \end{equation} where $C$ maps latent $\Phi,v$ to digital observables (clickthrough, dwell time, transitions). Modern platforms satisfy this empirically due to dense telemetry. \begin{theorem}[Capture Feasibility] A digital platform with dense telemetry has full observability of $(\Phi,\mathbf{v})$ up to harmonic null modes, enabling near-optimal capture control. \end{theorem} \begin{proof} Telemetry streams provide high-dimensional projections of both gradient and flow fields, satisfying the Popov–Belevitch–Hautus test with probability 1 in high dimension. $\square$ \end{proof}
