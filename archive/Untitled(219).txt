
\section{Hypergraph Catalysis and Nonlinear Spectra}
\begin{narrative}
Not all catalytic relations are pairwise.  Many ideas require multiple simultaneous premises—multi-meme reactions.  Hypergraph formalization extends the theory to these collective inferences.
\end{narrative}

Define hyperedges \(\mathcal{C}\subseteq M\) as catalyst sets.  
Reaction \(r_i\) occurs if \(\mathcal{C}\subseteq C(r_i)\), with rate  
\[
\dot{x}_i \propto \sum_{\mathcal{C}}A_{i,\mathcal{C}}\prod_{j\in\mathcal{C}}x_j^\alpha.
\]
Spectral theory generalizes using nonlinear Perron–Frobenius theorems for positive tensors; existence of persistent states corresponds to dominant tensor eigenvalue \(\rho_T(A)>\tau\).  

\section{Practical Simulation and Empirical Detection}
\begin{narrative}
To operationalize these models, adjacency matrices can be inferred from corpora using embedding similarity or conditional information metrics.  Dynamics then fit observed temporal sequences.
\end{narrative}

Estimate \(A_{ij}\) via cosine similarity or pointwise mutual information; fit parameters by minimizing
\[
\mathcal L = \sum_i\!\left[\dot{x}_i - \beta_i\sigma\!\left(\sum_jA_{ij}x_j^\alpha\right)-u_i+\lambda_i x_i\right]^2.
\]
Graph neural networks can learn approximate \(A_\theta\) from data, automating detection of emergent memeplexes.

\section{A1. Kinetic Derivation from Autocatalytic Closure}
\begin{narrative}
Memetic kinetics arise directly from closure conditions.  Each meme \(m_i\) is a catalytic transformation on interpretive substrate \(s_i\); the rate of change of its activation reflects both internal catalysis and external stimulation.
\end{narrative}

\[
\dot{x}_i = \sum_{r_i:\, i\in P(r_i)} 
\beta_i \!\!\!\!\!\!\!\!\!\!
\prod_{j\in C(r_i)}x_j^{\alpha_{ij}} 
+ \sum_{f\in F}\gamma_{if} - \lambda_i x_i .
\]
Linearization gives \(\dot{x}=(\beta cA-\lambda I)x+u\).
Stationarity requires \(\rho(A)>\lambda/(\beta c)\).

\section{A2. Entropy Balance and Boundary Flux}
\begin{narrative}
Entropy measures divergence from interpretive equilibrium \(q\).  
The correct balance law must include boundary terms, especially in open cultural systems with inflow of novelty or external information.
\end{narrative}

Discrete KL entropy:
\[
D_{\mathrm{KL}}(x\|q)=\sum_i x_i\log\frac{x_i}{q_i}.
\]
Time derivative:
\[
\dot D_{\mathrm{KL}}
=\sum_i\dot x_i(\log\frac{x_i}{q_i}+1).
\]
For the RSVP continuum:
\[
\partial_t(\rho\log\frac{\rho}{q})
=-\nabla\!\cdot\!(\rho\log\frac{\rho}{q}\mathbf v)
+\rho\nabla\!\cdot\!\mathbf v
+(\mathcal{R}-\Lambda)\log\frac{\rho}{q}.
\]
Integration over domain \(\Omega\):
\[
\frac{d}{dt}\!\int_\Omega\rho\log\frac{\rho}{q}
=\!\int_\Omega(\Pi-\Phi_{\text{diss}})\,d\mathbf r
+\!\oint_{\partial\Omega}\!\rho\log\frac{\rho}{q}\,\mathbf v\!\cdot\!d\mathbf S.
\]
The last term quantifies epistemic exchange with the external world.

\section{A3. Spectral and Lyapunov Stability}
\begin{narrative}
Spectral criteria ensure existence of steady states; Lyapunov functions confirm their stability.
\end{narrative}

Linearized \(\dot x=Gx\), \(G=c\,\mathrm{diag}(\beta)A-\mathrm{diag}(\lambda)\).
A positive fixed point exists if the dominant eigenvalue of \(G\) is positive.
Define \(V=\tfrac12\|x-x^*\|^2\); then
\[
\dot V=(x-x^*)^\top(G-G^*)(x-x^*),
\]
negative definite when \(\lambda_i>c\beta_i\rho(A)\).

Field energy functional
\[
E=\int_\Omega(\Phi^2+S^2+\rho^2)\,d\mathbf r
\]
satisfies
\[
\dot E=-\int_\Omega(\kappa_\Phi\Phi^2+\kappa_SS^2+\lambda\rho^{p+1})\,d\mathbf r
+\oint_{\partial\Omega}\rho\mathbf v\!\cdot\!d\mathbf S.
\]

\section{A4. Global Sector Stability}
\begin{narrative}
Local linear analysis omits nonlinear saturation.  Bounded sigmoids guarantee global sector stability under Lipschitz reaction terms.
\end{narrative}

Assume \(\sigma(z)\in[0,1]\) and 
\(|\mathcal F_\Phi|\le L_\Phi|\Phi|\), 
\(|\mathcal F_S|\le L_S|S|\).  
Then
\[
\dot V \le -\sum_i(\lambda_i - c\beta_iL_\Phi)\,(x_i-x_i^*)^2,
\]
yielding global asymptotic stability when \(\lambda_i>c\beta_iL_\Phi\).  
In field form,
\[
\dot E \le -\!\int_\Omega\!(\kappa_\Phi-L_\Phi)\Phi^2
-(\kappa_S-L_S)S^2-(\lambda-D_\rho k^2)\rho^2,
\]
ensuring monotone energy decay under bounded nonlinearity.

\section{A5. Threshold Bounds via Harmonic Means}
\begin{narrative}
For heterogeneous systems, harmonic means give sharper persistence bounds than arithmetic or geometric ones.

\section{A6. Stochastic and Perturbative Extensions}
\begin{narrative}
Cultural diffusion includes stochastic fluctuations: chance invention, communication noise, and attention shifts.  These can be modeled as Langevin perturbations.
\end{narrative}

\[
\dot{x}_i = \beta_i\sigma\!\left(\sum_jA_{ij}x_j^\alpha\right)
+u_i-\lambda_i x_i+\xi_i(t),
\]
\(\langle\xi_i(t)\xi_j(t')\rangle=2D_i\delta_{ij}\delta(t-t')\).
The corresponding Fokker–Planck equation:
\[
\partial_t P =
-\sum_i\partial_{x_i}\!\left[(\beta_i\sigma(\phi_i)
+u_i-\lambda_i x_i)P\right]
+\sum_iD_i\partial_{x_i}^2P.
\]

Data-driven inference of catalytic structure is essential for testing memetic autocatalysis empirically.
