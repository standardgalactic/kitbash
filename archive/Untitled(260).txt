const c=document.getElementById('stars'), ctx=c.getContext('2d');
const p=document.getElementById('planets'), px=p.getContext('2d');
const modalWrap=document.getElementById('modalWrap');
const closeModalBtn=document.getElementById('closeModal');
const holoTitle=document.getElementById('holoTitle');
const holoContent=document.getElementById('holoContent');
const indicator=document.getElementById('cycleIndicator');

let w,h,stars=[],planets=[],t=0,currentCycleIndex=0;

function resize(){
  w=innerWidth; h=innerHeight;
  c.width=p.width=w; c.height=p.height=h;
  stars=Array.from({length:500},()=>({x:Math.random()*w,y:Math.random()*h,z:Math.random()*w}));
  planets=cycles.map((cy,i)=>({name:cy,r:30,orbit:150+100*i,angle:Math.random()*Math.PI*2,clr:`hsl(${120+10*i},100%,60%)`}));
}
addEventListener('resize',resize); resize();

function drawStars(){
  ctx.fillStyle='rgba(0,0,0,0.4)'; ctx.fillRect(0,0,w,h);
  for(const s of stars){ s.z-=2; if(s.z<=0)s.z=w;
    const k=128/s.z; const x=w/2+(s.x-w/2)*k; const y=h/2+(s.y-h/2)*k;
    ctx.fillStyle='#00ff7f'; ctx.fillRect(x,y,1.5,1.5);
  } requestAnimationFrame(drawStars);
}
drawStars();

function drawPlanets(){
  px.clearRect(0,0,w,h); t+=0.01;
  for(const planet of planets){
    const x=w/2+Math.cos(t+planet.angle)*planet.orbit;
    const y=h/2+Math.sin(t+planet.angle)*planet.orbit;
    px.beginPath(); px.arc(x,y,planet.r,0,Math.PI*2);
    px.strokeStyle=planet.clr; px.shadowBlur=15; px.shadowColor=planet.clr;
    px.stroke(); px.fillStyle=planet.clr; px.fillText(planet.name,x+planet.r+5,y);
    planet.x=x; planet.y=y;
  }
  requestAnimationFrame(drawPlanets);
}
drawPlanets();

function openCycle(cycleName){
  animateCamera(()=>{ holoTitle.textContent=cycleName; holoContent.innerHTML='';
    BOOKS.filter(b=>b.cycle===cycleName).forEach(b=>{
      const card=document.createElement('div'); card.className='book-card';
      card.innerHTML=`<div class='book-title'>${b.n}. ${b.title}</div>
                      <div class='book-abstract'>${b.abstract}</div>`;
      holoContent.appendChild(card);
    });
    modalWrap.style.display='block'; updateIndicator();
  });
}

p.addEventListener('click',e=>{
  const rect=p.getBoundingClientRect(); const mx=e.clientX-rect.left; const my=e.clientY-rect.top;
  for(const planet of planets){
    const dx=mx-planet.x, dy=my-planet.y;
    if(Math.sqrt(dx*dx+dy*dy)<planet.r){
      currentCycleIndex=cycles.indexOf(planet.name); openCycle(planet.name); return;
    }
  }
});

closeModalBtn.addEventListener('click',()=>modalWrap.style.display='none');
document.getElementById('prevCycle').onclick=()=>navigate(-1);
document.getElementById('nextCycle').onclick=()=>navigate(1);
window.onkeydown=e=>{
  if(['ArrowLeft','a','A'].includes(e.key))navigate(-1);
  if(['ArrowRight','d','D'].includes(e.key))navigate(1);
};

function navigate(d){
  currentCycleIndex=(currentCycleIndex+d+cycles.length)%cycles.length;
  openCycle(cycles[currentCycleIndex]);
}
function updateIndicator(){
  indicator.textContent=`Cycle ${currentCycleIndex+1} of ${cycles.length} — ${cycles[currentCycleIndex]}`;
}
function animateCamera(cb){
  c.style.opacity='0.5'; p.style.transform='scale(0.9)';
  setTimeout(()=>{p.style.transform='scale(1.1)';
    setTimeout(()=>{p.style.transform='scale(1)'; c.style.opacity='1'; cb();},700);
  },300);
}
