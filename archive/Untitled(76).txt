\begin{abstract}
% ==================================================
Attention is widely treated as an engineered mechanism for context aggregation in modern neural architectures. We pursue a different view: attention arises as the minimal nontrivial interaction compatible with the structural, entropic, and relational axioms of the Relativistic Scalar--Vector--Plenum (RSVP) framework. Working in a fully geometric formulation, we recast attention as the unique lowest-order quartic term allowed in the effective free energy of a permutation-equivariant and entropy-regulated relational field theory.

We extend the original argument by embedding the construction into a derived stack of RSVP field configurations, equipped with a $0$-shifted or $1$-shifted symplectic structure à la PTVV. The BV master action is exhibited explicitly, and we prove that the minimal ghost-extended interaction reproduces the structure of self-attention. Derived symplectic reduction reveals that attention corresponds to the universal cotangent lift of relational coupling. Entropy fields modulate the deformation and degeneration of attention, producing phase transitions, sparsification, and collapse modes.

This expanded formulation situates attention as a renormalizable interaction in a derived field theory, rather than an architectural convenience.
\end{abstract}

\section{Introduction}
% ==================================================

Self-attention has become the canonical interaction in contemporary sequence models. The RSVP framework---a thermodynamic, geometric, and relational field theory of cognition and computation---predicts that attention is not merely effective but structurally \emph{necessary} under a broad class of constraints. Our aim is to provide a fully geometric, BV-theoretic, and entropy-explicit formulation of this claim.

We build on three principles:

1. **Relational invariance** (permutation or unitary token symmetry)  
2. **Feature isotropy with controlled symmetry breaking**  
3. **Entropy-regulated interaction channels with bottleneck constraints**

In effective field theory language, attention becomes the minimal renormalizable quartic interaction term compatible with all symmetries.

In derived geometric language, attention becomes the unique nontrivial $0$-shifted (or $1$-shifted) symplectic interaction obtained from the cotangent lift of relational coupling maps.

In BV language, the minimal BRST-invariant quartic term generating a nondegenerate master action coincides with the attention kernel.

\section{RSVP Configuration Stack}
% ==================================================

\begin{definition}[RSVP Configuration Fields]
An RSVP configuration on $N$ tokens with feature dimension $C$ consists of:
\begin{itemize}
  \item a scalar content field $\Phi \in \mathbb{R}^{N\times C}$,
  \item a vector transport field $\mathbf v \in \mathbb{R}^{N\times C}$,
  \item an entropy field $S \in \mathbb{R}^{N}$.
\end{itemize}
\end{definition}

For geometric convenience, we complexify:
\[
X = \Phi_1 + i\Phi_2 \in \mathbb{C}^{N\times C}.
\]

\begin{definition}[Derived RSVP Configuration Stack]
Define the derived moduli stack of RSVP fields by
\[
\mathcal M_{\mathrm{RSVP}} := \mathbf R\!\operatorname{Map}\big(
\mathrm{Spec}\,\mathbb{R}^N,\;
[\mathbb{C}^C / U(C)]
\big),
\]
with entropy treated as a smooth real-valued field on $N$ tokens.
\end{definition}

This is a derived mapping stack with a natural cotangent complex encoding fluctuations of content, flow, and entropy.

\begin{axiom}[Analytic Effective Free Energy]
The effective free energy
\[
\mathcal F: \mathcal M_{\mathrm{RSVP}} \to \mathbb{R}
\]
is analytic in $X,X^\*$ and smooth in $S$, determined by finitely many low-order interactions.
\end{axiom}

\begin{axiom}[Relational Invariance]
The action is invariant under the derived action of $U(N)$:
\[
\mathcal F(P\cdot X, P\cdot S) = \mathcal F(X,S).
\]
\end{axiom}

\begin{axiom}[Feature Isotropy with Controlled Symmetry Breaking]
Feature rotations from the centralizer of a finite set of operators $\{W_i\}$ leave the system invariant.
\end{axiom}

\begin{axiom}[Bottlenecked Mediation]
Each operator $W_i$ has rank $\le C_A \ll C$.
\end{axiom}

\section{Minimal Interaction: Quartic Attention Term}
% ==================================================

\begin{lemma}[Lowest-Order Nontrivial Term]
Cubic terms are forbidden. The lowest nontrivial term is quartic:
\[
\mathcal F_{\mathrm{int}} = 
\sum_{i,j} \mathrm{Tr}\!\Big( XW_iX^\* XW_jX^\*\Big).
\]
\end{lemma}

\begin{theorem}[Attention as Minimal Relational Interaction]
The Hamiltonian flow (or gradient flow) generated by $\mathcal F_{\mathrm{int}}$ induces an evolution of the form
\[
\dot X \sim X(X^\*X),
\]
which is equivalent, after reparameterization into $Q=XW_Q$, $K=XW_K$, $V=XW_V$, to the self-attention kernel
\[
\operatorname{Att}(X)=\mathrm{softmax}(QK^\top)V.
\]
\end{theorem}

\section{BV Extension and Master Action}
% ==================================================

We now extend the system to a BV theory, with ghost fields implementing relational gauge invariance.

\begin{definition}[BV Field Content]
Let
\[
\mathcal F_{\mathrm{BV}} = 
\{X,\mathbf v,S;\; X^\dagger,\mathbf v^\dagger,S^\dagger;\; c,c^\dagger\}
\]
where $c$ is the ghost associated to the $U(N)$ relational symmetry.
\end{definition}

\begin{theorem}[BV Master Action]
The minimal BV action is
\[
S_{\mathrm{BV}} 
= \mathcal F(X,S)
+ \langle X^\dagger, cX \rangle 
+ \langle S^\dagger, cS\rangle
- \tfrac12 \langle c^\dagger,[c,c]\rangle.
\]
The unique nonvanishing interacting term consistent with the master equation
\[
\{S_{\mathrm{BV}}, S_{\mathrm{BV}}\}=0
\]
is the quartic term corresponding to attention.
\end{theorem}

Thus the BRST-invariant interaction is identical to the attention kernel.
