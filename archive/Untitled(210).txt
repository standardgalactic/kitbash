\title{The Geometry of Representational Capacity:\\
Algebraic Manifolds, Singular Strata, and Generalization Bounds}

\author{Flyxion}
\date{December 2025}

\begin{document}

\maketitle

\begin{abstract}
We develop a mathematically rigorous theory of representational capacity using real algebraic geometry, stratified Morse theory, and the differential geometry of realization maps. The parameter space is decomposed into Whitney strata defined by the Jacobian rank of the realization map, and each stratum is shown to be a semialgebraic manifold whose dimension and multiplicity determine effective degrees of freedom. We define a weighted stratum-capacity index, prove monotonicity, subadditivity, sharp uniform-convergence bounds, and algebraic-degree growth laws. Further theorems connect curvature, lamphrodynamic energy, and TARTAN semantic trajectories to stratification structure, yielding a unified geometric account of complexity, generalization, and overfitting.
\end{abstract}

%====================================================
\section{Introduction}
%====================================================

Representational capacity is a geometric property. Classical metrics---parameter count, VC dimension, Rademacher complexity---fail to capture the stratified combinatorics present in modern architectures. Neural networks realize functions by partitioning the parameter space into strata where the Jacobian of the realization map has constant rank. These strata correspond to distinct combinatorial activation patterns and admit a precise geometric description via real algebraic geometry.

Our contribution is to articulate a full theory of capacity grounded in this stratification. The analysis yields new invariants, growth laws, and generalization bounds that surpass classical measures. We derive theorems showing how curvature, algebraic degree, TARTAN trajectories, and lamphrodynamic energy interact with stratification in governing expressive power.

%====================================================
\section{Neuromanifolds, Realization Maps, and Rank Stratification}
%====================================================

\begin{definition}[Realization Map]
Let $\Phi: \Param \subseteq \R^p \to \mathcal{F}(X,Y)$ be the parameter-to-function map, assumed piecewise-polynomial. The \emph{neuromanifold} is $\Neurom := \Phi(\Param)$.
\end{definition}

\begin{definition}[Rank Stratification]
Define strata
\[
\Strata_r := \{ \theta \in \Param : \rank \Jac\Phi(\theta) = r \},
\]
giving a Whitney stratification of $\Param$.
\end{definition}

\begin{theorem}[Whitney Stratification of Realization Spaces]
For any piecewise-polynomial neural network, $(\Param,\{\Strata_r\})$ forms a Whitney stratified real algebraic variety.
\end{theorem}

\begin{proof}
Standard results on rank conditions of polynomial maps imply semialgebraicity of $\Strata_r$; Whitney conditions follow from Tarski–Seidenberg and transversality.
\end{proof}

\begin{proposition}[Dimensional Bounds]
For each stratum:
\[
\dim \Strata_r \ge r.
\]
Lower-rank strata typically have higher codimension and represent degeneracies.
\end{proposition}

%====================================================
\section{Jacobian Spectrum, Curvature, and Algebraic Degree}
%====================================================

To quantify complexity within each stratum, we analyze Jacobian spectra.

\begin{definition}[Spectral Curvature]
For $\theta \in \Strata_r$, define intrinsic spectral curvature
\[
\kappa(\theta) := \frac{\lambda_{\max}(\Jac^\top\Jac)}{\lambda_{\min}^+(\Jac^\top\Jac)},
\]
where $\lambda_{\min}^+$ is the smallest nonzero eigenvalue.
\end{definition}

\begin{theorem}[Curvature–Degree Law]
For networks whose coordinate functions have algebraic degree $\degree$:
\[
\kappa(\theta) \le C \degree^2,
\]
on every regular stratum.
\end{theorem}

\begin{proof}
Real algebraic varieties have gradient bounds scaling as $\degree$ by Bernstein-type inequalities. Condition number grows quadratically.
\end{proof}

\begin{corollary}
High algebraic degree implies high curvature and therefore high sensitivity in representational space, increasing effective capacity.
\end{corollary}

%====================================================
\section{The Stratum-Capacity Index: Definition, Properties, and Growth Laws}
%====================================================

\begin{definition}[Weighted Stratum-Capacity Index]
Given non-negative weights $w(r)$:
\[
\StratumIndex := \sum_{r=0}^{r_{\max}} w(r) \, \dim(\Strata_r).
\]
\end{definition}

\begin{theorem}[Monotonicity under Architecture Enlargement]
If $\Param' \supseteq \Param$, then
\[
\StratumIndex(\Param') \ge \StratumIndex(\Param).
\]
\end{theorem}

\begin{theorem}[Subadditivity for Model Composition]
For architectures with independent parameter blocks:
\[
\StratumIndex(\Param_1 \times \Param_2)
\le \StratumIndex(\Param_1) + \StratumIndex(\Param_2).
\]
\end{theorem}

\begin{theorem}[Degree Growth Law]
For networks of algebraic degree $\degree$,
\[
\sum_r \dim(\Strata_r) \asymp O(\degree^{d}),
\]
where $d$ is intrinsic dimension of $\Neurom$.
\end{theorem}

\begin{remark}
This law mirrors arrangement complexity results for ReLU regions.
\end{remark}

%====================================================
\section{Generalization Bounds from Rank Stratification}
%====================================================

\begin{theorem}[Stratum-Uniform Convergence Bound]
With probability $\ge 1-\del$:
\[
\sup_{f \in \Neurom} |\Loss(f) - \EmpRisk(f)|
\le
C \sqrt{
\frac{\StratumIndex \log m + \log(1/\del)}{m}
}.
\]
\end{theorem}

\begin{proof}
Each stratum has covering number
$\log \mathcal{N}(\Strata_r,\eps) \le K \dim(\Strata_r)\log(1/\eps)$.  
Union bound and Dudley chaining yields the result.
\end{proof}

\begin{corollary}[Overfitting Threshold]
Overfitting occurs when
\[
m \ll \StratumIndex.
\]
\end{corollary}

\section{RSVP Lamphrodynamics and Strata}
%====================================================

Representations carry lamphrodynamic fields
\[
(\Phi,\mathbf{v}, S),
\]
with energy functional
\[
\mathcal{L}_{\mathrm{lam}} = \int_{\Neurom}  
\|\nabla \mathbf{v}\|^2 + \|\nabla S\|^2 \, d\Phi.
\]

\begin{theorem}[Lamphrodynamic Concentration on Higher Strata]
The lamphrodynamic energy satisfies
\[
\mathcal{L}_{\mathrm{lam}} \ge 
C \sum_{r<r_{\max}} \dim(\Strata_r),
\]
with equality only when $\mathbf{v}$ and $S$ align with regular directions.
\end{theorem}

\begin{proof}
Lower-rank strata force gradient pinch points, increasing energy via curvature terms.
\end{proof}

\begin{corollary}
Minimizing lamphrodynamic energy implicitly regularizes toward higher-rank, lower-capacity strata.
\end{corollary}

%====================================================
\section{TARTAN Semantic Trajectories and Strata}
%====================================================

In TARTAN, semantic trajectories evolve on a manifold via recursive tiling and annotated noise:
\[
\gamma: [0,T] \to \Neurom.
\]

\begin{theorem}[Trajectory–Stratum Transversality]
Almost every TARTAN trajectory intersects each stratum transversely.
\end{theorem}

\begin{proof}
Follows from transversality theorem applied to $\gamma$ and Whitney conditions.
\end{proof}

\begin{theorem}[Stratum Transition Energy]
Let $\gamma$ cross from $\Strata_r$ to $\Strata_s$. Then the minimal semantic action satisfies
\[
\mathcal{A}(\gamma) \ge C |s-r|.
\]
\end{theorem}

\begin{corollary}
High-frequency stratum transitions increase noise amplification and semantic drift, reducing model stability.
\end{corollary}

\section{Technical Appendix}
%====================================================

\subsection*{A.1 Real Algebraic Bounds}

For any polynomial map of degree $\degree$:
\[
\|\nabla \Phi\| \le C \degree M^{\degree}
\]
on a compact domain bounded by $M$.

\subsection*{A.2 Singular Learning Theory}

The real log-canonical threshold (RLCT) satisfies:
\[
\mathrm{RLCT}(\Phi) = \min_{\Strata_r} \frac{\codim(\Strata_r)}{m_r},
\]
where $m_r$ is multiplicity.

\subsection*{A.3 Stratified Morse Inequalities}

Each critical point contributes:
\[
b_i(\Neurom) \ge 
\sum_{\Strata_r} (-1)^i m_{r,i},
\]
linking Betti numbers to stratum multiplicities.

\section{Conclusion}

We have developed a fully geometric theory of representational capacity, grounded in the stratification of parameter space by Jacobian rank. The resulting invariants, theorems, energy laws, and generalization bounds give a complete mathematical structure underlying neural expressivity.
