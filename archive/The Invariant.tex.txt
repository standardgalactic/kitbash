\actheading{Act I --- Dartmouth, 1956}
% ============================================================

\sceneheading{INT. CONFERENCE ROOM --- MORNING}

\action{Chalkboard. Coffee thermos. Ashtray smoke.}

\action{McCARTHY is arranging papers. MINSKY is pacing. SHANNON sits with a pad of graph paper.}

\action{The door opens.}

\action{HARTMANN enters.}

\action{He carries a single matte black briefcase.}

\action{No one recognizes him.}

\character{McCarthy}
\dialog{You're here for the AI workshop?}

\character{Hartmann}
\dialog{I am.}

\action{He sets the briefcase gently on the long wooden table.}

\action{It looks out of place. Too clean. No scuffs. No brand.}

\character{Minsky}
\dialog{Planning to take notes?}

\character{Hartmann}
\dialog{Something like that.}

\action{He clicks two latches.}

\action{The case opens.}

\action{Inside: not paper.}

\action{A precisely machined internal frame unfolds.}

\action{Metal arms extend silently, locking into position.}

\action{A wide, flat panel rises from the center --- not a screen, not glowing --- just a tensioned membrane with embedded mechanical components.}

\action{Spools emerge from recessed compartments.}

\action{Thin tape threads itself through a reading head with uncanny smoothness.}

\action{Shannon leans forward.}

\character{Shannon}
\dialog{That's not Ampex.}

\character{Hartmann}
\dialog{No.}

\character{McCarthy}
\dialog{Is it a recorder?}

\character{Hartmann}
\dialog{It records and predicts.}

\action{Minsky laughs.}

\character{Minsky}
\dialog{Predicts what?}

\character{Hartmann}
\dialog{Continuation.}

\action{The machine hums softly.}

\action{No digital sounds. No beeps. Just mechanical precision.}

\action{Hartmann speaks into it:}

\character{Hartmann}
\dialog{Define object.}

\action{The tape spins.}

\action{The machine pauses.}

\action{Then it plays back in Hartmann's own voice:}

\character{The Machine}
\paren{Hartmann's voice}
\dialog{An object is that which maintains structural invariance under admissible deformation.}

\action{The room goes still.}

\character{Minsky}
\dialog{That's your voice.}

\character{Hartmann}
\dialog{Yes.}

\character{McCarthy}
\dialog{It's playing back what you said.}

\character{Hartmann}
\dialog{No.}

\action{The machine speaks again:}

\character{The Machine}
\paren{Hartmann's voice}
\dialog{If invariance fails, identity degrades.}

\action{Silence.}

\action{Shannon stands, walks around the table, examining the device without touching it.}

\character{Shannon}
\dialog{What's the encoding basis?}

\character{Hartmann}
\dialog{Entropy.}

\character{Minsky}
\dialog{That's absurd.}

\character{Hartmann}
\dialog{No. It's mechanical.}

\action{The tape continues moving.}

\action{The device is not magical. It is physical. Gears. Tension. Constraint.}

\action{It looks like it could have been built in 1956.}

\action{Except it couldn't have.}

\character{McCarthy}
\dialog{What does it assume?}

\character{Hartmann}
\dialog{That identity depends on continuation.}

\action{The machine emits a faint clicking sound --- almost like a ratchet.}

\character{Hartmann}
\dialog{It cannot rewind beyond stability.}

\action{Shannon looks at the ratcheting mechanism.}

\character{Shannon}
\dialog{It's irreversible.}

\character{Hartmann}
\dialog{Yes.}

\character{Hale}
\paren{from across the room}
\dialog{So if I define a heart logically, your machine disagrees?}

\character{Hartmann}
\dialog{It tests persistence.}

\character{Hale}
\dialog{Against what?}

\character{Hartmann}
\dialog{Future possibility.}

\action{The tape tightens.}

\action{The machine stops.}

\action{No one speaks.}

\character{Minsky}
\dialog{If that's real, it means axioms aren't enough.}

\action{Hartmann closes the machine slowly.}

\action{The frame collapses with quiet precision.}

\action{The spools retract.}

\action{The membrane folds.}

\action{In seconds, it is a simple briefcase again.}

\character{McCarthy}
\dialog{You built that?}

\character{Hartmann}
\dialog{No.}

\action{That line hangs.}

\action{Shannon watches Hartmann carefully now.}

\action{He understands something the others do not.}

% ============================================================
\sceneheading{INT. CONFERENCE ROOM --- LATE AFTERNOON}
% ============================================================

\action{McCarthy stands at the board. Clean handwriting.}

\action{He writes: OBJECT / PROPERTY / RELATION / RULE}

\character{McCarthy}
\dialog{If we define objects and their properties precisely, reasoning becomes mechanical. The world can be described in predicates. Intelligence is inference over representation.}

\action{He circles ``OBJECT.''}

\character{McCarthy}
\dialog{We begin with stable entities. A person. A block. A number. Assign properties. State relations. Apply rules.}

\character{Newell}
\dialog{And if the rules are correct, the system reaches correct conclusions.}

\character{Minsky}
\dialog{The problem is scale, not principle.}

\character{Hale}
\dialog{We must get the ontology right first. If we misclassify the kinds of things that exist, reasoning fails.}

\action{McCarthy nods.}

\character{McCarthy}
\dialog{Exactly. Intelligence begins with correct kinds.}

\action{The room feels unified.}

\action{Hartmann has been listening quietly.}

\character{Hartmann}
\dialog{And how do you know when you've chosen the correct kinds?}

\character{Minsky}
\dialog{They reflect the world.}

\character{Hartmann}
\dialog{The world when?}

\action{That doesn't land yet.}

\character{McCarthy}
\dialog{The world as it is.}

\action{Hartmann walks slowly to the board.}

\character{Hartmann}
\dialog{Suppose we define a chair as an object with four legs and a seat.}

\character{Minsky}
\paren{smirks}
\dialog{Then we refine the definition.}

\character{Hartmann}
\dialog{Suppose the chair loses a leg.}

\character{Newell}
\dialog{It becomes a broken chair.}

\character{Hartmann}
\dialog{And if I stack books under the missing leg?}

\character{Hale}
\dialog{It is a repaired chair.}

\character{Hartmann}
\dialog{And if I turn it upside down and use it as a ladder?}

\character{Minsky}
\dialog{Then you're misusing it.}

\character{Hartmann}
\dialog{Or redefining it.}

\action{The room shifts slightly.}

\character{McCarthy}
\dialog{These are linguistic ambiguities. The ontology just needs to encode use-cases.}

\character{Hartmann}
\dialog{No. These are stability cases.}

\action{He picks up the chalk.}

\character{Hartmann}
\dialog{You are treating objects as definitions. But in practice, we treat them as things that survive abuse.}

\character{Minsky}
\dialog{Survive abuse?}

\character{Hartmann}
\dialog{Yes. A chair is not what satisfies a predicate. It's what still works after someone stands on it wrong.}

\action{A few laughs.}

\character{Hartmann}
\dialog{An object is something that continues to function despite perturbation.}

\character{Hale}
\dialog{That's engineering, not ontology.}

\character{Hartmann}
\dialog{It's both.}

\character{Hartmann}
\dialog{When my son was five, he called every four-legged animal a ``dog.'' That was his ontology. It worked until he met a cow.}

\action{Light laughter.}

\character{Hartmann}
\dialog{He didn't refine his definition because of axioms. He refined it because reality pushed back.}

\action{Shannon looks up from his notes.}

\character{Hartmann}
\dialog{You're designing systems that assume the categories are correct before they're tested. But in practice, categories are tested by resistance.}

\character{McCarthy}
\dialog{So what are you proposing?}

\character{Hartmann}
\dialog{That instead of beginning with objects and attaching rules, we begin with constraints and ask which patterns persist.}

\character{Minsky}
\dialog{That's far more complicated.}

\character{Hartmann}
\dialog{No. It's more honest.}

\character{Hale}
\dialog{You're turning ontology into fluid.}

\character{Hartmann}
\dialog{No. I'm saying objects are the stable parts of the fluid.}

\action{That line lands harder.}

\character{McCarthy}
\dialog{You're saying the object is secondary.}

\character{Hartmann}
\dialog{I'm saying the object is what survives.}

\action{Pause.}

\character{Newell}
\dialog{And how would you implement that?}

\character{Hartmann}
\dialog{You test stability before classification.}

\action{The room grows quiet.}

\action{This is the fork.}

\newpage

% ============================================================
\sceneheading{INT. CONFERENCE ROOM --- CONTINUOUS}
% ============================================================

\action{The board still reads: OBJECT / PROPERTY / RELATION / RULE}

\action{Underneath, Hartmann has drawn a crooked little chair.}

\character{Minsky}
\dialog{You're romanticizing repair. We're building reasoning systems, not furniture.}

\character{Hartmann}
\dialog{No. I'm pointing out that your system assumes the chair exists before it is stressed.}

\character{McCarthy}
\dialog{Stress is a separate problem. First define the object.}

\character{Hartmann}
\dialog{And what if stress defines the object?}

\character{Hale}
\dialog{That makes identity conditional.}

\character{Hartmann}
\dialog{It already is.}

\character{Hale}
\dialog{A chair does not cease to be a chair because a leg is broken.}

\character{Hartmann}
\dialog{It might. If it collapses when sat upon.}

\action{A few people nod despite themselves.}

\character{Newell}
\dialog{We'd encode states. Whole chair. Broken chair. Repaired chair.}

\character{Hartmann}
\dialog{And how many states before the definition fails?}

\action{He pauses.}

\character{Hartmann}
\dialog{You're planning to enumerate exceptions. I'm suggesting you detect failure.}

\character{Minsky}
\dialog{Failure of what?}

\character{Hartmann}
\dialog{Failure of persistence.}

\action{Shannon finally stands.}

\character{Shannon}
\dialog{Persistence relative to what?}

\character{Hartmann}
\dialog{To expected use.}

\character{Shannon}
\dialog{Expected use is a distribution.}

\action{The room quiets. That word lands differently.}

\action{Shannon walks to the board and adds a new column: EXPECTED FUTURES}

\character{Shannon}
\dialog{If a chair admits a narrow set of future states --- sitting, holding weight --- it has low uncertainty. If it admits anything, it's not a chair.}

\character{McCarthy}
\dialog{So you're measuring classification by admissible continuation.}

\character{Shannon}
\dialog{Yes.}

\character{Hale}
\dialog{But that makes ontology probabilistic.}

\character{Shannon}
\dialog{It makes it operational.}

\character{Hartmann}
\dialog{You are trying to describe what a thing is independent of what it can become.}

\character{Minsky}
\dialog{And you're saying that's impossible.}

\character{Hartmann}
\dialog{I'm saying it's misleading.}

\character{Newell}
\dialog{So instead of declaring ``This is a chair,'' the system asks, ``Does this pattern support sitting?''}

\character{Hartmann}
\dialog{Yes.}

\character{McCarthy}
\dialog{That's function, not form.}

\character{Hartmann}
\dialog{Function is what tests form.}

\character{Hale}
\paren{shakes his head}
\dialog{You're dissolving categories into behavior.}

\character{Hartmann}
\dialog{No. I'm grounding them in constraint.}

\action{Silence again.}

\character{Minsky}
\dialog{Let's make this concrete. Suppose we define ``person'' as an object with properties A, B, C. You're saying that's insufficient.}

\character{Hartmann}
\dialog{Yes.}

\character{Minsky}
\dialog{What's missing?}

\character{Hartmann}
\dialog{The fact that a person tomorrow is not identical to a person today unless something persists across change.}

\character{McCarthy}
\dialog{That's identity through time. Philosophy has handled that for centuries.}

\character{Hartmann}
\dialog{Philosophy assumes the identity. I'm asking how you detect it.}

\character{Shannon}
\dialog{If two states look identical but admit different futures, they aren't equivalent.}

\action{Now the room is fully engaged.}

\character{Newell}
\dialog{So our systems shouldn't just infer. They should test stability.}

\character{Minsky}
\dialog{Test how?}

\character{Hartmann}
\dialog{Perturb the model. See what survives.}

\action{McCarthy studies the word OBJECT on the board.}

\action{He walks up slowly.}

\action{He draws a line through it.}

\action{He writes underneath: PATTERN}

\character{Minsky}
\dialog{Pattern is vague.}

\character{McCarthy}
\dialog{Object was premature.}

\character{Hale}
\dialog{You're surrendering realism.}

\character{McCarthy}
\dialog{No. We're postponing it.}

\action{Shannon quietly adds one more word beneath PATTERN: CONSTRAINT}

\action{The board now reads:}

\action{\hspace{1in}PATTERN\\
\hspace{1in}CONSTRAINT\\
\hspace{1in}CONTINUATION}

\action{No one says ``entropy.''\\
No one says ``homotopy.''\\
But the direction has changed.}

\newpage

% ============================================================
\sceneheading{INT. CONFERENCE ROOM --- LATER}
% ============================================================

\action{Hale stands slowly. He does not raise his voice.}

\character{Hale}
\dialog{You are confusing engineering prudence with ontology.}

\action{The room stills.}

\character{Hale}
\dialog{A bridge must survive stress. A chair must support weight. That does not make ``bridge'' or ``chair'' reducible to perturbation tests.}

\character{Hartmann}
\dialog{It makes them accountable to reality.}

\character{Hale}
\dialog{No. It makes them conditional on circumstance.}

\action{He turns to the room.}

\character{Hale}
\dialog{A child calling cows ``dogs'' does not prove that kinds are unstable. It proves that children are ignorant.}

\action{Light laughter.}

\character{Hale}
\dialog{Your proposal treats ontology as something discovered only after stress. But classification must precede stress. You must know what something is before you can test it.}

\action{He looks at McCarthy.}

\character{Hale}
\dialog{If we don't begin with objects, what exactly is being perturbed?}

\action{That lands.}

\action{Minsky nods slightly.}

\character{Hale}
\dialog{You speak of persistence under change. But persistence of what?}

\action{He walks to the board and writes over PATTERN: SUBJECT}

\character{Hale}
\dialog{There must be a bearer. A thing that has properties. A heart does not become a ``non-heart'' simply because it fails.}

\character{Hartmann}
\dialog{If it cannot circulate blood, what is it?}

\character{Hale}
\dialog{A defective heart.}

\character{Hartmann}
\dialog{Then defect is already part of the ontology.}

\character{Hale}
\dialog{Yes. But it is parasitic on the original category.}

\action{He turns back to the room.}

\character{Hale}
\dialog{If identity is earned, then every object exists on probation. That is not realism. That is instability.}

\action{He sharpens it.}

\character{Hale}
\dialog{Suppose your system encounters a dying patient. Entropy rising, constraint collapsing. Does the person cease to be a person?}

\action{Silence.}

\action{Hartmann does not answer immediately.}

\character{Hale}
\dialog{Your framework risks eroding dignity in the name of dynamical honesty.}

\action{The room tightens.}

\action{McCarthy looks between them.}

\action{Minsky folds his arms.}

\action{Shannon watches carefully.}

\action{Hartmann finally speaks.}

\character{Hartmann}
\dialog{I am not saying identity disappears when constraint weakens.}

\character{Hale}
\dialog{But you are saying it depends on constraint.}

\character{Hartmann}
\dialog{Yes.}

\character{Hale}
\dialog{Then identity is never secure.}

\character{Hartmann}
\dialog{Neither is life.}

\action{That line is quiet. Not triumphant.}

\character{Hale}
\dialog{Science cannot operate on metaphysical fragility. It requires fixed reference.}

\action{He gestures at the board.}

\character{Hale}
\dialog{``Person'' must mean the same thing today and tomorrow. Otherwise databases collapse, reasoning collapses, law collapses.}

\character{Hale}
\dialog{If ontology floats on continuation, then two identical states could be different merely because their futures differ. That is absurd.}

\action{The room leans forward.}

\action{This is the technical strike.}

\character{Hartmann}
\dialog{They are different.}

\character{Hale}
\dialog{In what way?}

\character{Hartmann}
\dialog{In the possibilities they permit.}

\character{Hale}
\dialog{Possibility is not being.}

\character{Shannon}
\paren{softly}
\dialog{In communication systems, it is.}

\character{Hale}
\dialog{We are not building telephones.}

\character{Shannon}
\dialog{No. You're building systems that must distinguish states. If two states look identical but behave differently under perturbation, your machine must treat them differently.}

\character{Hale}
\dialog{That is engineering discrimination, not ontological revision.}

\character{Hartmann}
\dialog{Perhaps ontology should answer to engineering.}

\action{Hale shakes his head.}

\character{Hale}
\dialog{You are replacing essence with behavior.}

\character{Hartmann}
\dialog{I am replacing assumption with test.}

\action{Long pause.}

\action{McCarthy studies the board again.}

\action{Hale gathers his notes but does not leave.}

\character{Hale}
\dialog{If you proceed this way, you will never have a clean system. Everything will depend on context.}

\character{Hartmann}
\dialog{Yes.}

\character{Hale}
\dialog{Then your intelligence will hesitate.}

\character{Hartmann}
\dialog{It will survive.}

\sceneheading{INT. CONFERENCE ROOM --- EVENING}
% ============================================================

\action{Tension still in the air.}

\character{Minsky}
\paren{impatient}
\dialog{Enough hypotheticals. Let's test this.}

\action{He walks to the blackboard.}

\character{Minsky}
\dialog{Let's define something simple.}

\action{He writes: BLOCK / ON(X,Y) / CLEAR(X)}

\action{McCarthy smiles faintly. This is familiar ground.}

\character{McCarthy}
\dialog{A block world. Stable objects. Clear relations. We can reason perfectly in it.}

\character{Newell}
\paren{nods}
\dialog{We prove the system.}

\character{Hale}
\paren{folds arms}
\dialog{Yes. Exactly.}

\action{Hartmann watches.}

\character{Minsky}
\dialog{Block A is on Block B. B is on the table. Clear means nothing on top. Simple.}

\action{He turns to Hartmann.}

\character{Minsky}
\dialog{Where's the instability?}

\action{Hartmann walks forward slowly.}

\character{Hartmann}
\dialog{May I?}

\action{He erases nothing. He adds one word to the board: BREAKABLE(A)}

\action{A few eyebrows rise.}

\character{Hartmann}
\dialog{Suppose Block A is brittle.}

\character{Minsky}
\dialog{So what?}

\character{Hartmann}
\dialog{You stack B on A.}

\character{McCarthy}
\dialog{Then ON(B,A).}

\character{Hartmann}
\dialog{Yes. Now suppose A shatters.}

\action{Silence.}

\character{Newell}
\dialog{Then A no longer satisfies BLOCK.}

\character{Hartmann}
\dialog{Exactly.}

\character{Hale}
\dialog{It becomes fragments. Still objects.}

\character{Hartmann}
\dialog{But your system did not anticipate that ON(B,A) could invalidate the identity of A.}

\character{McCarthy}
\dialog{That's just incomplete modeling.}

\character{Hartmann}
\dialog{No. It's structural.}

\action{He continues calmly.}

\character{Hartmann}
\dialog{You begin with objects and assume relations act on them. But relations can destroy the object.}

\action{The room shifts.}

\character{Minsky}
\dialog{Then we include fragility as a property.}

\character{Hartmann}
\dialog{And how many such properties before your ontology becomes a catalog of exceptions?}

\character{McCarthy}
\dialog{Finite many.}

\character{Hartmann}
\dialog{You don't know that.}

\action{Hartmann turns to Shannon.}

\character{Hartmann}
\dialog{Claude, may I borrow that branching diagram?}

\action{Shannon nods.}

\action{Hartmann draws two identical squares labeled A1 and A2.}

\character{Hartmann}
\dialog{Two blocks. Identical properties. Same predicates. Same relations.}

\action{He draws B stacked on each.}

\character{Hartmann}
\dialog{Now imagine A1 is internally cracked. A2 is not. Your ontology cannot see the difference.}

\character{Hale}
\dialog{Then your ontology is incomplete.}

\character{Hartmann}
\dialog{Yes. But how do you complete it?}

\character{McCarthy}
\dialog{You refine the description.}

\character{Hartmann}
\dialog{How far?}

\action{Silence.}

\character{Hartmann}
\dialog{You will always be behind reality. Because the difference is not in the current state. It's in what happens next.}

\character{Shannon}
\dialog{If stacking produces collapse in one case but not the other, then they are not equivalent states.}

\character{McCarthy}
\dialog{Equivalent under current predicates.}

\character{Shannon}
\dialog{But not under continuation.}

\action{That word again.}

\character{Newell}
\dialog{So you're saying classification must account for potential.}

\character{Hartmann}
\dialog{Yes. Otherwise two identical descriptions produce different outcomes.}

\character{Minsky}
\dialog{Then intelligence must simulate perturbations.}

\character{Hartmann}
\dialog{Exactly.}

\action{Now it lands.}

\action{This is not philosophy. This is engineering foresight.}

\character{Hale}
\dialog{You are expanding ontology into prediction.}

\character{Hartmann}
\dialog{No. I'm refusing to pretend ontology is independent of prediction.}

\action{McCarthy stares at the two blocks on the board.}

\character{McCarthy}
\paren{slowly}
\dialog{If our system cannot distinguish A1 from A2 before stacking, it will misreason.}

\character{Hartmann}
\dialog{Yes.}

\character{McCarthy}
\dialog{So identity must encode stability conditions.}

\character{Shannon}
\dialog{Or at least admit uncertainty.}

\character{Newell}
\dialog{That complicates inference.}

\character{Hartmann}
\dialog{It prevents catastrophe.}

\action{The room is very quiet now. No one laughs. No one storms out.}

\action{This is the pivot.}

\character{Hale}
\paren{one final attempt}
\dialog{Then your system must treat everything as fragile.}

\character{Hartmann}
\dialog{No. Only what has not yet been tested.}

\action{That's the line.}

\action{The engine hums softly.}

\character{McCarthy}
\dialog{It still looks the same.}

\character{Hartmann}
\dialog{It should.}

\action{McCarthy runs a hand across the dash.}

\character{McCarthy}
\dialog{People said we'd live in glass towers. Talking to machines.}

\character{Hartmann}
\dialog{We do talk to machines.}

\character{McCarthy}
\dialog{Not like that.}

\character{Hartmann}
\dialog{No.}

\sceneheading{EXT. UNIVERSITY PARKING LOT --- CONTINUOUS}

\action{They pull in.}

\action{The lot is full of similar vehicles. All mid-century designs. All clean. All quiet.}

\action{No one seems rushed.}

\action{Hartmann turns off the ignition.}

\action{Nothing dramatic happens.}

\action{He steps out.}

\action{McCarthy steps out.}

\action{Hartmann walks to the front of the car.}

\action{He rests his hand flat against the hood.}

\action{There is no sound. No glow.}

\action{The car's surface subtly loosens, as if joints release tension.}

\action{The chassis folds inward along lines you never noticed before.}

\action{The chrome compresses. The wheels retract without spinning. The glass becomes opaque and then metallic.}

\action{In under five seconds, the entire car compacts into a matte rectangular case the size of a large briefcase.}

\action{No spectacle. No flashing light. Just precise mechanical reconfiguration.}

\action{Hartmann lifts it easily.}

\action{McCarthy watches, still not entirely used to this.}

\character{McCarthy}
\dialog{It still feels wrong.}

\character{Hartmann}
\dialog{Why?}

\character{McCarthy}
\dialog{It's a car.}

\character{Hartmann}
\dialog{It's a configuration.}

\action{They begin walking toward the building.}

\newpage

\sceneheading{INT. UNIVERSITY CORRIDOR --- MOMENTS LATER}

\action{Scholars pass them. No one reacts to the briefcase.}
