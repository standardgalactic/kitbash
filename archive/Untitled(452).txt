\chapter{Functional Calculus for Metric-Dependent Quantities}

This appendix summarizes the functional derivative tools required to compute
the Quantum Affine Shift Tensor (\QAST). We work on a $d$-dimensional
pseudo-Riemannian manifold $(M,g)$ with coordinates $x^\mu$.

\section{Variation of the Inverse Metric}

The inverse metric satisfies $g_{\mu\alpha} g^{\alpha\nu} = \delta_\mu^{\;\nu}$.
Variation yields
\begin{equation}
\delta g^{\mu\nu}
=
- g^{\mu\alpha} g^{\nu\beta} \, \delta g_{\alpha\beta}.
\label{eq:inverse-metric-variation}
\end{equation}

Thus the functional derivative is
\begin{equation}
\frac{\delta g^{\mu\nu}(x)}{\delta g_{\alpha\beta}(y)}
=
- g^{\mu\alpha}(x) g^{\nu\beta}(x) \delta^{(d)}(x-y).
\label{eq:inverse-metric-func-der}
\end{equation}

\section{Variation of the Christoffel Symbols of the First Kind}

The quantity
\[
\Gamma_{\sigma,\nu\rho}
=
\frac{1}{2} \left(
\partial_\nu g_{\rho\sigma}
+ \partial_\rho g_{\nu\sigma}
- \partial_\sigma g_{\nu\rho}
\right)
\]
varies linearly with the metric:
\begin{equation}
\delta \Gamma_{\sigma,\nu\rho}
=
\frac{1}{2} \left(
\nabla_\nu \delta g_{\rho\sigma}
+ \nabla_\rho \delta g_{\nu\sigma}
- \nabla_\sigma \delta g_{\nu\rho}
\right).
\label{eq:first-kind-variation}
\end{equation}

\section{Variation of the Christoffel Symbols of the Second Kind}

Using $\Gamma^\mu_{\nu\rho} = g^{\mu\sigma} \Gamma_{\sigma,\nu\rho}$,
we obtain:
\begin{equation}
\boxed{

=
\frac{1}{2} g^{\mu\sigma}
\left(
\nabla_\nu \delta g_{\rho\sigma}
+ \nabla_\rho \delta g_{\nu\sigma}

\label{eq:second-kind-variation}
\end{equation}

This expression is the fundamental object required for Appendix B.
